29  // Get F on T Axia
30  Flux GetFluxUpwind(const Primitive& left, const Primitive& right) const {
31  auto *non_const_this = const_cast<HartenLaxLeerContact *>(this);
32  non_const_this->Initialize(left, right);
33  Flux flux;
34  if (0.0 <= wave_left_) {
35  flux = GetFlux(left);
36  } else if (wave_right_ <= 0.0) {
37  flux = GetFlux(right);
38  } else if (0.0 <= wave_star_) {
39  flux = GetStarFlux(left, wave_left_);
40  } else if (wave_star_ < 0.0) {
41  flux = GetStarFlux(right, wave_right_);
42  }
43  return flux;
44  }
45  // Get F of U
46  static Flux GetFlux(const Primitive& state) {
47  return Gas::PrimitiveToFlux(state);
48  }
49 
50  private:
51  Speed wave_star_;
52  Speed wave_left_;
53  Speed wave_right_;
54  void Initialize(const Primitive& left, const Primitive& right) {
55  double rho_average = (left.rho() + right.rho()) / 2;
56  double a_left = Gas::GetSpeedOfSound(left);
57  double a_right = Gas::GetSpeedOfSound(right);
58  double a_average = (a_left + a_right) / 2;
59  double p_pvrs = (left.p() + right.p() - (right.u() - left.u()) *
60  rho_average * a_average) / 2;
61  double p_estimate = std::max(0.0, p_pvrs);
62  wave_left_ = left.u() - a_left * GetQ(p_estimate, left.p());
63  wave_right_ = right.u() + a_right * GetQ(p_estimate, right.p());
64  wave_star_ = (right.p() - left.p() +
65  left.rho() * left.u() * (wave_left_ - left.u()) -
66  right.rho() * right.u() * (wave_right_ - right.u())) /
67  (left.rho() * (wave_left_ - left.u()) -
68  right.rho() * (wave_right_ - right.u()));
69  }
70  static double GetQ(Scalar const& p_estimate, Scalar const& p_k) {
71  if (p_estimate <= p_k) {
72  return 1.0;
73  } else {
74  double temp = 1 + Gas::GammaPlusOneOverTwo() * (p_estimate / p_k - 1) /
75  Gas::Gamma();
76  return std::sqrt(temp);
77  }
78  }
79  void SetMomentum(Scalar rho, const Primitive &primitive,
80  Conservative *conservative) const requires(kDimensions == 1) {
81  conservative->momentumX() = rho * wave_star_;
82  }
83  void SetMomentum(Scalar rho, const Primitive &primitive,
84  Conservative *conservative) const requires(kDimensions == 2) {
85  conservative->momentumX() = rho * wave_star_;
86  conservative->momentumY() = rho * primitive.v();
87  }
88  void SetMomentum(Scalar rho, const Primitive &primitive,
89  Conservative *conservative) const requires(kDimensions == 3) {
90  conservative->momentumX() = rho * wave_star_;
91  conservative->momentumY() = rho * primitive.v();
92  conservative->momentumZ() = rho * primitive.w();
93  }
94  Flux GetStarFlux(const Primitive& state, Speed const& wave_k) const {
95  Flux flux = GetFlux(state);
96  Conservative u_k = Gas::PrimitiveToConservative(state);
97  double energy = u_k.energy();
98  double temp = state.rho() * (wave_k - state.u()) / (wave_k - wave_star_);
99  Conservative u_star_k;
100  u_star_k.mass() = temp;
101  SetMomentum(temp, state, &u_star_k);
102  u_star_k.energy() = energy / state.rho() + (wave_star_ - state.u()) *
103  (wave_star_ + state.p() / (state.rho() * (wave_k - state.u())));
104  u_star_k.energy() *= temp;
105  u_star_k -= u_k;
106  u_star_k *= wave_k;
107  flux += u_star_k;
108  return flux;
109  }
