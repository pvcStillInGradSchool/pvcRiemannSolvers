59  assert(residual);

70  void AddFluxToHolderAndSharer(Face const &face,
71  Scalar *holder_data, Scalar *sharer_data) const override {
72  const auto &riemanns = this->GetRiemannSolvers(face);
73  const auto &integrator = face.integrator();
74  const auto &holder = face.holder();
75  const auto &sharer = face.sharer();
76  for (int q = 0, n = integrator.CountPoints(); q < n; ++q) {
77  const auto &coord = integrator.GetGlobal(q);
78  Value u_holder = holder.GlobalToValue(coord);
79  Value u_sharer = sharer.GlobalToValue(coord);
80  Value flux = riemanns[q].GetFluxUpwind(u_holder, u_sharer);
81  flux *= -integrator.GetGlobalWeight(q);
82  Coeff prod = flux * holder.GlobalToBasisValues(coord);
83  assert(holder_data);
84  Polynomial::AddToResidual(prod, holder_data);
85  if (nullptr == sharer_data) { continue; }
86  prod = -flux * sharer.GlobalToBasisValues(coord);
87  Polynomial::AddToResidual(prod, sharer_data);
88  }
