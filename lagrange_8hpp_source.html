34 template <std::floating_point S, int kDegree>
35 class Line {
36  public:
37  static constexpr int P = kDegree; // the degree of each member in this basis
38  static constexpr int N = P + 1; // the number of terms in this basis
39  using Scalar = S;
40  using Vector = algebra::Matrix<Scalar, 1, N>;
41  using Matrix = algebra::Matrix<Scalar, N, N>;
42  using Taylor = basis::Taylor<Scalar, 1, kDegree>;
43 
44  private:
48  std::array<std::array<Vector, N>, N> derivatives_;
49  Vector nodes_; // nodes_[j] := coorinate of the j-th node
50  Matrix lagrange_to_taylor_;
51 
52  public:
53  template <std::ranges::random_access_range R>
54  explicit Line(R &&nodes) {
55  assert(nodes.size() == N);
56  /* An arbitrary polynomial can be expanded on Taylor and Lagrange:
57  * polynomial(x) = taylor_basis_row(x) @ taylor_coeff_col
58  * = lagrange_basis_row(x) @ lagrange_coeff_col.
59  * We want to find `taylor_to_lagrange` and `lagrange_to_taylor` such that
60  * taylor_coeff_col = lagrange_to_taylor @ lagrange_coeff_col,
61  * and lagrange_coeff_col = taylor_to_lagrange @ taylor_coeff_col.
62  * Then, from
63  * taylor_basis_row(x) @ taylor_coeff_col =
64  * taylor_basis_row(x) @ lagrange_to_taylor @ lagrange_coeff_col =
65  * lagrange_basis_row(x) @ lagrange_coeff_col,
66  * We have
67  * taylor_basis_row(x) @ lagrange_to_taylor = lagrange_basis_row(x),
68  * and lagrange_basis_row(x) @ taylor_to_lagrange = taylor_basis_row(x).
69  * Since lagrange_basis_row(x[j])[i] = delta[i][j], we have
70  * taylor_to_lagrange[j] := values of Taylor basis at the j-th node.
71  */
72  Matrix taylor_to_lagrange;
73  for (int j = 0; j < N; ++j) {
74  auto x_j = (nodes_[j] = nodes[j]);
75  taylor_to_lagrange.row(j) = Taylor::GetValues(x_j);
76  }
77  lagrange_to_taylor_ = taylor_to_lagrange.inverse();
78  assert((taylor_to_lagrange * lagrange_to_taylor_
79  - Matrix::Identity()).norm() < 1e-12);
80  for (int k = 0; k < N; ++k) {
81  for (int j = 0; j < N; ++j) {
82  auto x_j = nodes_[j];
83  derivatives_[k][j] = GetDerivatives(k, x_j);
84  }
85  }
86  }

89  : Line(std::views::iota(nodes.begin(), nodes.end())
90  | std::views::transform([](auto *ptr){ return *ptr; })) {
91  }

100  return nodes_[i];

109  Vector GetValues(Scalar x) const {
110  Vector vec; vec.setOnes();
111  for (int i = 0; i < N; ++i) {
112  auto x_i = nodes_[i];
113  for (int j = 0; j < N; ++j) {
114  if (i != j) {
115  auto x_j = nodes_[j];
116  vec[i] *= (x - x_j) / (x_i - x_j);
117  }
118  }
119  }
120  return vec;
121  }

130  Vector GetDerivatives(int k, Scalar x) const {
131  Vector taylor_basis_row = Taylor::GetDerivatives(k, x);
132  return taylor_basis_row * lagrange_to_taylor_;
133  }
