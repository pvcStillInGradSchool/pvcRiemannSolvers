61  // Get U on t-Axis
62  Primitive GetPrimitiveUpwind(const Primitive& left,
63  const Primitive& right) {
64  Primitive result;
65  if (left.rho() > 0) {
66  if (right.rho() > 0) {
67  // ordinary case
68  } else {
69  // only right vacuum
70  auto a_left = Gas::GetSpeedOfSound(left);
71  if (left.u() >= a_left) {
72  result = left;
73  } else {
74  auto u_star = left.u() + a_left * Gas::GammaMinusOneUnderTwo();
75  if (u_star <= 0) {
76  result.setZero();
77  } else {
78  auto gri_1 = left.p() / std::pow(left.rho(), Gas::Gamma());
79  result = PrimitiveInsideExpansion(gri_1, u_star);
80  }
81  }
82  return result;
83  }
84  } else {
85  if (right.rho() > 0) {
86  // only left vacuum
87  auto a_right = Gas::GetSpeedOfSound(right);
88  if (right.u() + a_right <= 0) {
89  result = right;
90  } else {
91  auto u_star = right.u() - a_right * Gas::GammaMinusOneUnderTwo();
92  if (u_star >= 0) {
93  result.setZero();
94  } else {
95  auto gri_1 = right.p() / std::pow(right.rho(), Gas::Gamma());
96  result = PrimitiveInsideExpansion(gri_1, u_star);
97  }
98  }
99  } else {
100  // both vacuum
101  result.setZero();
102  }
103  return result;
104  }
105  // Construct the function of speed change, aka the pressure function.
106  auto u_change_given = right.u() - left.u();
107  auto u_change__left = SpeedChange(left);
108  auto u_change_right = SpeedChange(right);
109  auto f = [&](double p) {
110  return u_change__left(p) + u_change_right(p) + u_change_given;
111  };
112  auto f_prime = [&](double p) {
113  return u_change__left.Prime(p) + u_change_right.Prime(p);
114  };
115  if (f(0) < 0) { // Ordinary case: Wave[2] is a contact.
116  Primitive star; // state in star region
117  star.p() = FindRoot(f, f_prime, left.p());
118  star.u() = 0.5 * (right.u() + u_change_right(star.p())
119  +left.u() - u_change__left(star.p()));
120  star_u = star.u();
121  if (0 < star.u()) { // Axis[t] <<< Wave[2]
122  if (star.p() >= left.p()) { // Wave[1] is a shock.
123  result = PrimitiveNearShock<1>(left, &star);
124  } else { // star.p() < left.p() : Wave[1] is an expansion.
125  result = PrimitiveNearExpansion<1>(left, &star);
126  }
127  } else { // star.u() < 0 : Wave[2] <<< Axis[t] <<< Wave[3].
128  if (star.p() >= right.p()) { // Wave[3] is a shock.
129  result = PrimitiveNearShock<3>(right, &star);
130  } else { // star.p() < right.p() : Wave[3] is an expansion.
131  result = PrimitiveNearExpansion<3>(right, &star);
132  }
133  }
134  } else { // The region BETWEEN Wave[1] and Wave[3] is vacuumed.
135  result = PrimitiveNearVacuum(left, right);
136  }
137  return result;
138  }
139  // Helper method and class for the star region:
140  template <class F, class Fprime>
141  static double FindRoot(F&& f, Fprime&& f_prime, double x, double eps = 1e-8) {
142  while (f(x) > 0) {
143  x *= 0.5;
144  }
145  while (f(x) < -eps) {
146  auto divisor = f_prime(x);
147  assert(divisor != 0);
148  x -= f(x) / divisor;
149  }
150  assert(std::abs(f(x)) < eps);
151  return x;
152  }
153  class SpeedChange {
154  public:
155  explicit SpeedChange(const Primitive& before)
156  : rho_before_(before.rho()), p_before_(before.p()),
157  a_before_(before.rho() == 0 ? 0 : Gas::GetSpeedOfSound(before)),
158  p_const_(before.p() * Gas::GammaMinusOneOverTwo()) {
159  assert(before.rho() >= 0 && before.p() >= 0);
160  }
161  double operator()(double p_after) const {
162  double value;
163  assert(p_after >= 0);
164  if (p_after >= p_before_) { // shock
165  auto divisor = std::sqrt(rho_before_ * P(p_after));
166  assert(divisor != 0);
167  value = (p_after - p_before_) / divisor;
168  } else { // expansion
169  constexpr auto exp = Gas::GammaMinusOneOverTwo() / Gas::Gamma();
170  value = std::pow(p_after / p_before_, exp) - 1;
171  value *= Gas::GammaMinusOneUnderTwo() * a_before_;
172  }
173  return value;
174  }
175  double Prime(double p_after) const {
176  assert(p_after >= 0);
177  double value;
178  if (p_after >= p_before_) { // shock
179  auto p = P(p_after);
180  value = Gas::GammaPlusOneOverFour() * (p_before_ - p_after);
181  auto divisor = std::sqrt(rho_before_ * p);
182  assert(divisor != 0);
183  value = (1 + value / p) / divisor;
184  } else { // expansion
185  constexpr double exp = -Gas::GammaPlusOneOverTwo() / Gas::Gamma();
186  auto divisor = rho_before_ * a_before_;
187  assert(divisor != 0);
188  value = std::pow(p_after / p_before_, exp) / divisor;
189  }
190  return value;
191  }
192 
193  private:
194  double P(double p_after) const {
195  return Gas::GammaPlusOneOverTwo() * p_after + p_const_;
196  }
197 
198  private:
199  double rho_before_, p_before_, a_before_, p_const_;
200  };

220  template <int kField>
221  static Primitive PrimitiveNearShock(const Primitive& before,
222  Primitive* after) {
223  static_assert(kField == 1 || kField == 3);
224  auto shock = Shock<kField>(before, *after);
225  if (TimeAxisAfterShock(shock)) { // i.e. (x=0, t) is AFTER the shock.
226  after->rho() = before.rho();
227  auto divisor = after->u() - shock.u;
228  assert(divisor != 0);

230  return *after;

235  // Expansion and related methods:
236  template <int kField>
237  class Expansion {
238  static constexpr double AddOrMinus(double x, double y) {
239  static_assert(kField == 1 || kField == 3);
240  return kField == 1 ? (x + y) : (x - y);
241  }
242 
243  public:
244  double a_before, a_after; // speed of sound before/after the wave
245  double gri_1, gri_2; // Generalized Riemann Invariants
246  Expansion(const Primitive& before, const Primitive& after)
247  : a_before(Gas::GetSpeedOfSound(before)),
248  gri_1(before.p() / (std::pow(before.rho(), Gas::Gamma()))) {
249  gri_2 = AddOrMinus(before.u(),
250  Gas::GammaMinusOneUnderTwo() * a_before);
251  a_after = AddOrMinus(a_before,
252  Gas::GammaMinusOneOverTwo() * (before.u() - after.u()));
253  }
254  };
255  static Primitive PrimitiveInsideExpansion(double gri_1, double gri_2) {
256  constexpr auto r = Gas::GammaMinusOne() / Gas::GammaPlusOne();
257  auto a = r * gri_2;
258  auto a_square = a * a;
259  auto rho = std::pow(a_square / gri_1 * Gas::OneOverGamma(),
260  Gas::OneOverGammaMinusOne());
261  Primitive axis; // state on t-axis
262  axis.rho() = rho;
263  axis.u() = a;
264  axis.p() = a_square * rho * Gas::OneOverGamma();
265  return axis;
266  }
267  template <int kField>
268  constexpr static bool TimeAxisAfterExpansion(double u, double a) {
269  static_assert(kField == 1 || kField == 3);
270  return kField == 1 ? (u - a < 0) : (u + a > 0);
271  }
272  template <int kField>
273  static bool TimeAxisBeforeExpansion(double u, double a) {
274  static_assert(kField == 1 || kField == 3);
275  return kField == 1 ? (u - a > 0) : (u + a < 0);
276  }
277  template <int kField>
278  static Primitive PrimitiveNearExpansion(const Primitive& before,
279  Primitive* after) {
280  static_assert(kField == 1 || kField == 3);
281  auto wave = Expansion<kField>(before, *after);
282  if (TimeAxisAfterExpansion<kField>(after->u(), wave.a_after)) {
283  auto divisor = wave.a_after * wave.a_after;
284  assert(divisor != 0);
285  after->rho() = Gas::Gamma() * after->p() / divisor;
286  return *after;
287  } else if (TimeAxisBeforeExpansion<kField>(before.u(), wave.a_before)) {
288  return before;
289  } else { // Axis[t] is inside the Expansion.
290  return PrimitiveInsideExpansion(wave.gri_1, wave.gri_2);
291  }
292  }
293  static Primitive PrimitiveNearVacuum(const Primitive& left,
294  const Primitive& right) {
295  auto left_a = Gas::GetSpeedOfSound(left);
296  if (left.u() > left_a) { // Axis[t] <<< Wave[1].
297  return left;
298  }
299  auto right_a = Gas::GetSpeedOfSound(right);
300  if (right.u() + right_a < 0) { // Wave[3] <<< Axis[t].
301  return right;
302  } else { // Wave[1] <<< Axis[t] <<< Wave[3].
303  auto gri_1 = left.p() / (std::pow(left.rho(), Gas::Gamma()));
304  auto gri_2 = left.u() + left_a * Gas::GammaMinusOneUnderTwo();
305  if (gri_2 >= 0) { // Axis[t] is inside Wave[1].
306  return PrimitiveInsideExpansion(gri_1, gri_2);
307  } else { // gri_2 < 0
308  // Axis[t] is to the RIGHT of Wave[1].
309  gri_1 = right.p() / (std::pow(right.rho(), Gas::Gamma()));
310  gri_2 = right.u() - right_a * Gas::GammaMinusOneUnderTwo();
311  if (gri_2 < 0) { // Axis[t] is inside Wave[3].
312  return PrimitiveInsideExpansion(gri_1, gri_2);
313  } else { // Axis[t] is inside the vacuumed region.
314  Primitive axis; // state on t-axis
315  axis.setZero();
316  return axis;
317  }
318  }
319  }
