157  Local LocalToJacobianDeterminantGradient(const Local &xyz) const {
158  Local det_grad;
159  auto mat_grad = LocalToJacobianGradient(xyz);
160  Jacobian mat = LocalToJacobian(xyz);

174  Hessian LocalToJacobianDeterminantHessian(const Local &xyz) const {
175  Hessian det_hess;
176  auto mat_hess = LocalToJacobianHessian(xyz);
177  Jacobian mat = LocalToJacobian(xyz);
178  Jacobian inv = mat.inverse();
179  det_hess[XX] = (inv * mat_hess[XX]).trace();
180  det_hess[XY] = (inv * mat_hess[XY]).trace();
181  det_hess[XZ] = (inv * mat_hess[XZ]).trace();
182  det_hess[YY] = (inv * mat_hess[YY]).trace();
183  det_hess[YZ] = (inv * mat_hess[YZ]).trace();
184  det_hess[ZZ] = (inv * mat_hess[ZZ]).trace();
185  auto mat_grad = LocalToJacobianGradient(xyz);
186  mat_grad[X] = inv * mat_grad[X];
187  mat_grad[Y] = inv * mat_grad[Y];
188  mat_grad[Z] = inv * mat_grad[Z];
189  auto trace_x = mat_grad[X].trace();
190  auto trace_y = mat_grad[Y].trace();
191  auto trace_z = mat_grad[Z].trace();
192  det_hess[XX] += trace_x * trace_x - (mat_grad[X] * mat_grad[X]).trace();
193  det_hess[XY] += trace_x * trace_y - (mat_grad[X] * mat_grad[Y]).trace();
194  det_hess[XZ] += trace_x * trace_z - (mat_grad[X] * mat_grad[Z]).trace();
195  det_hess[YY] += trace_y * trace_y - (mat_grad[Y] * mat_grad[Y]).trace();
196  det_hess[YZ] += trace_y * trace_z - (mat_grad[Y] * mat_grad[Z]).trace();
197  det_hess[ZZ] += trace_z * trace_z - (mat_grad[Z] * mat_grad[Z]).trace();
198  det_hess *= mat.determinant();
199  return det_hess;

236 void SortNodesOnFace(const Cell<Scalar> &cell, const T *cell_nodes,
237  U *face_nodes, int face_n_node) {
238  size_t cell_nodes_copy[64], face_nodes_copy[32];

246  }
247  if (sizeof(U) == sizeof(size_t)) {
248  face_node_list = reinterpret_cast<size_t *>(const_cast<U *>(face_nodes));
249  } else {
250  face_node_list = face_nodes_copy;
251  std::copy_n(face_nodes, face_n_node, face_node_list);
252  }

255  if (sizeof(U) != sizeof(size_t)) {
