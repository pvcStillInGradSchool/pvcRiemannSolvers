26  EigenMatrices() = default;
27  EigenMatrices(const Primitive& primitive, // orthonormal vectors:
28  const Mat3x1& nu, const Mat3x1& mu, const Mat3x1& pi) {
29  constexpr int x = 0, y = 1, z = 2;
30  auto u_x = primitive.u(), u_y = primitive.v(), u_z = primitive.w();
31  auto u_nu = (u_x * nu[x] + u_y * nu[y] + u_z * nu[z]);
32  auto u_mu = (u_x * mu[x] + u_y * mu[y] + u_z * mu[z]);
33  auto u_pi = (u_x * pi[x] + u_y * pi[y] + u_z * pi[z]);
34  auto a = IdealGas::GetSpeedOfSound(primitive);
35  auto &velocity = primitive.momentum();
36  auto ek = velocity.dot(velocity) / 2;
37  auto e0 = a * a * IdealGas::OneOverGammaMinusOne();
38  auto h0 = e0 + ek;
39  // build Mat5x5 R
40  R.col(0) << 1, u_x - a*nu[x], u_y - a*nu[y], u_z - a*nu[z], h0 - a*u_nu;
41  R.col(1) << 1, u_x, u_y, u_z, ek;
42  R.col(2) << 0, mu[x], mu[y], mu[z], u_mu;
43  R.col(3) << 0, pi[x], pi[y], pi[z], u_pi;
44  R.col(4) << 1, u_x + a*nu[x], u_y + a*nu[y], u_z + a*nu[z], h0 + a*u_nu;
45  // build Mat5x5 L
46  auto b1 = 1 / e0, b2 = b1 * ek;
47  L.col(0) << (b2 + u_nu / a) / 2, 1 - b2, -u_mu, -u_pi,
48  (b2 - u_nu / a) / 2;
49  L.col(1) << -(b1 * u_x + nu[x] / a) / 2, b1 * u_x, mu[x], pi[x],
50  -(b1 * u_x - nu[x] / a) / 2;
51  L.col(2) << -(b1 * u_y + nu[y] / a) / 2, b1 * u_y, mu[y], pi[y],
52  -(b1 * u_y - nu[y] / a) / 2;
53  L.col(3) << -(b1 * u_z + nu[z] / a) / 2, b1 * u_z, mu[z], pi[z],
54  -(b1 * u_z - nu[z] / a) / 2;
55  L.col(4) << b1 / 2, -b1, 0, 0, b1 / 2;
56  }
57  EigenMatrices(const Conservative &conservative, // orthonormal vectors:
58  const Mat3x1& nu, const Mat3x1& mu, const Mat3x1& pi) {
59  auto primitive = IdealGas::ConservativeToPrimitive(conservative);
60  *this = EigenMatrices(primitive, nu, mu, pi);
61  }
62  EigenMatrices(const Mat5x1 &tuple, const Mat3x3& frame) {
63  const auto &consv = *reinterpret_cast<const Conservative *>(&tuple);
64  *this = EigenMatrices(consv, frame.col(0), frame.col(1), frame.col(2));
65  }
