100  using Index = _Index<1>;

108  static Vector GetValues(Scalar x) {
109  Vector vec;
110  Scalar x_power = 1;
111  vec[0] = x_power;
112  for (int k = 1; k < N; ++k) {
113  vec[k] = (x_power *= x);
114  }
115  assert(std::abs(x_power - std::pow(x, P)) < 1e-14);
116  return vec;
117  }
118 
126  static Vector GetDerivatives(int k, Scalar x) {
127  assert(0 <= k && k <= P);
128  Vector vec;
129  vec.setZero(); // For all j < k, there is vec[j] = 0.
130  auto factorial_j = std::tgamma(Scalar(k + 1)); // factorial(j == k)
131  auto factorial_j_minus_k = Scalar(1); // factorial(j - k == 0)
132  // j * (j - 1) * ... * (j - k + 1) =
133  vec[k] = factorial_j / factorial_j_minus_k;
134  auto x_power = Scalar(1);
135  for (int j = k + 1; j < N; ++j) {
136  auto j_minus_k = j - k;
137  factorial_j_minus_k *= j_minus_k;
138  factorial_j *= j;
139  x_power *= x;
140  vec[j] = x_power * factorial_j / factorial_j_minus_k;
141  }
142  assert(std::abs(x_power - std::pow(x, P - k)) < 1e-14);
143  assert(factorial_j == std::tgamma(P + 1));
144  assert(factorial_j_minus_k == std::tgamma(P - k + 1));
145  return vec;
146  }
147 };
