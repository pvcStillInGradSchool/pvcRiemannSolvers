85  int face_n_node/* number of nodes on quadrangle */) {
86  int cnt = 0, c = 0, c_sum = 0;
87  while (cnt < 3) {
88  auto curr_node = cell_nodes[c];
89  for (int f = 0; f < face_n_node; ++f) {
90  if (face_nodes[f] == curr_node) {
91  c_sum += c;

102  case 5:

248  static void LocalToNewShapeHessians(Local const &local, Hessian *hessians) {
249  Scalar x_local = local[X], y_local = local[Y], z_local = local[Z];
250  // local_coords_[i][X] = local_coords_[i][Y] = local_coords_[i][Z] = 0

254  // shapes[26] = quadratic_x * quadratic_y * quadratic_z;

260  hessian[YY] = -2 * quadratic_x * quadratic_z;

271  Scalar quadratic_y = (1 - y_local * y_local);
272  Scalar quadratic_z = (1 - z_local * z_local);
273  // shapes[26] = quadratic_x * quadratic_y * quadratic_z;
274  Scalar linear_x = 4 * x_local;
275  Scalar linear_y = 4 * y_local;
276  Scalar linear_z = 4 * z_local;
277  auto &tensor = tensors[26];
278  tensor[XXX] = tensor[YYY] = tensor[ZZZ] = 0;
279  tensor[XXY] = linear_y * quadratic_z;
280  tensor[XXZ] = linear_z * quadratic_y;
281  tensor[YYX] = linear_x * quadratic_z;
282  tensor[YYZ] = linear_z * quadratic_x;
283  tensor[ZZX] = linear_x * quadratic_y;
284  tensor[ZZY] = linear_y * quadratic_x;
285  tensor[XYZ] = linear_x * linear_y * linear_z / (-8);
286  }
