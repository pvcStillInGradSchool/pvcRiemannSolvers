46  def get_spatial_matrix(self, scheme: spatial.FiniteElement, kappa_h: float):
47  """Get the spatial matrix of a FiniteElement scheme.
48  """
49  assert isinstance(scheme, spatial.FiniteElement)
50  # kappa_h = k_int * 2 * np.pi / scheme.length() * scheme.delta_x(0)
51  n_term = scheme.degree() + 1
52  matrices = np.zeros((self._n_element_n_element, n_term, n_term), dtype=complex)
53  for col in range(n_term):
54  u_tilde = np.zeros(n_term)
55  u_tilde[col] = 1
56  global_column = np.ndarray(scheme.n_dof(), dtype=complex)
57  for i_element in range(scheme.n_element()):
58  first = i_element * n_term
59  last = first + n_term
60  global_column[first:last] = u_tilde * np.exp(1j * i_element * kappa_h)
61  scheme.set_solution_column(global_column)
62  global_column = scheme.get_residual_column()
63  for i_element in range(scheme.n_element()):
64  matrix = matrices[i_element]
65  assert matrix.shape == (n_term, n_term)
66  first = i_element * n_term
67  last = first + n_term
68  matrix[:, col] = global_column[first:last]
69  matrix[:, col] /= np.exp(1j * i_element * kappa_h)
70  return matrices[-1]

72  def get_spatial_matrix_fast(self, scheme: spatial.FiniteElement, kappa_h: float):
73  """Get the spatial matrix of a FiniteElement scheme.
74 
75  This version is fast since it uses the fact that
76  r_curr(u_curr, u_prev, u_next) = s_curr * u_curr
77  + s_prev * u_prev
78  + s_next * u_next
79  """
80  assert isinstance(scheme, spatial.FiniteElement)
81  # kappa_h = k_int * 2 * np.pi / scheme.length() * scheme.delta_x(0)
82  n_term = scheme.degree() + 1
83  i_curr = self._n_element_n_element // 2
84  i_prev = i_curr - 1
85  i_next = i_curr + 1
86  s_curr = np.zeros((n_term, n_term), dtype=complex)
87  s_prev = np.zeros((n_term, n_term), dtype=complex)
88  s_next = np.zeros((n_term, n_term), dtype=complex)
89  first = i_curr * n_term
90  last = first + n_term
91  global_column = np.ndarray(scheme.n_dof(), dtype=complex)
92  for col in range(n_term):
93  # build s_curr[:, col] from residual_j(|k⟩, |0⟩, |0⟩)
94  global_column[:] = 0
95  global_column[i_curr * n_term + col] = 1
96  scheme.set_solution_column(global_column)
97  global_column = scheme.get_residual_column()
98  s_curr[:, col] = global_column[first:last]
99  # build s_prev[:, col] from residual_j(|0⟩, |k⟩, |0⟩)
100  global_column[:] = 0
101  global_column[i_prev * n_term + col] = 1
102  scheme.set_solution_column(global_column)
103  global_column = scheme.get_residual_column()
104  s_prev[:, col] = global_column[first:last]
105  # build s_next[:, col] from residual_j(|0⟩, |0⟩, |k⟩)
106  global_column[:] = 0
107  global_column[i_next * n_term + col] = 1
108  scheme.set_solution_column(global_column)
109  global_column = scheme.get_residual_column()
110  s_next[:, col] = global_column[first:last]
111  s_curr += s_prev * np.exp(-1j * kappa_h)
112  s_curr += s_next * np.exp(+1j * kappa_h)
113  return s_curr

135  def get_physical_mode(self, sampled_wavenumbers: np.ndarray,
136  modified_wavenumbers: np.ndarray):
137  n_sample, n_term = modified_wavenumbers.shape
138  assert n_sample == len(sampled_wavenumbers)
139  physical_eigvals = np.ndarray(n_sample, dtype=complex)
140  for i_sample in range(n_sample):
141  eigvals = modified_wavenumbers[i_sample]
142  # sort the eigvals by their norms
143  norms = eigvals.real**2 + eigvals.imag**2
144  pairs = np.ndarray(n_term, dtype=[('x', complex), ('y', float)])
145  for i_term in range(n_term):
146  pairs[i_term] = (eigvals[i_term], norms[i_term])
147  pairs.sort(order='y')
148  i_interval = int(np.floor(sampled_wavenumbers[i_sample] / np.pi))
149  while i_interval > n_term:
150  i_interval -= n_term * 2
151  while i_interval <= -n_term:
152  i_interval += n_term * 2
153  assert -n_term < i_interval <= n_term
154  if i_interval < 0:
155  i_mode = -1 - i_interval
156  else:
157  i_mode = i_interval - (i_interval == n_term)
158  assert 0 <= i_mode < n_term
159  physical_eigvals[i_sample] = pairs[i_mode][0]
160  return physical_eigvals
