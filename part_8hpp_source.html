581  > ShareGhostNodes(std::ifstream &istrm) {
582  char line[kLineWidth];
583  // send nodes info
584  std::map<Int, std::vector<Int>> send_nodes;
585  while (istrm.getline(line, kLineWidth) && line[0] != '#') {
586  int i_part, m_node;
587  std::sscanf(line, "%d %d", &i_part, &m_node);
588  send_nodes[i_part].emplace_back(m_node);
589  }
590  std::vector<MPI_Request> requests;
591  std::vector<std::vector<Scalar>> send_bufs;
592  for (auto &[i_part, nodes] : send_nodes) {
593  auto &coords = send_bufs.emplace_back();
594  for (auto m_node : nodes) {
595  auto [i_zone, i_node] = m_to_node_index_.at(m_node);
596  auto const &coord = GetCoord(i_zone, i_node);
597  coords.emplace_back(coord[0]);
598  coords.emplace_back(coord[1]);
599  coords.emplace_back(coord[2]);
600  }
601  assert(std::is_sorted(nodes.begin(), nodes.end()));
602  int n_reals = 3 * nodes.size();
603  int tag = i_part;
604  auto &request = requests.emplace_back();
605  MPI_Isend(coords.data(), n_reals, kMpiRealType, i_part, tag,
606  MPI_COMM_WORLD, &request);
607  }
608  // recv nodes info
609  std::map<Int, std::vector<Int>> recv_nodes;
610  while (istrm.getline(line, kLineWidth) && line[0] != '#') {
611  int i_part, m_node, i_zone, i_node;
612  std::sscanf(line, "%d %d %d %d", &i_part, &m_node, &i_zone, &i_node);
613  recv_nodes[i_part].emplace_back(m_node);
614  m_to_node_index_.emplace(m_node, NodeIndex(i_zone, i_node));
615  }
616  std::vector<std::vector<Scalar>> recv_coords;
617  for (auto &[i_part, nodes] : recv_nodes) {
618  assert(std::is_sorted(nodes.begin(), nodes.end()));
619  int n_reals = 3 * nodes.size();
620  auto &coords = recv_coords.emplace_back(std::vector<Scalar>(n_reals));
621  int tag = rank_;
622  auto &request = requests.emplace_back();
623  MPI_Irecv(coords.data(), n_reals, kMpiRealType, i_part, tag,
624  MPI_COMM_WORLD, &request);
625  }
626  // wait until all send/recv finish
627  std::vector<MPI_Status> statuses(requests.size());
628  MPI_Waitall(requests.size(), requests.data(), statuses.data());
629  requests.clear();
630  return { recv_nodes, recv_coords };
631  }

828  auto ShareGhostCells(GhostAdj const &ghost_adj) {
829  auto &send_npes = ghost_adj.send_npes;
830  auto &recv_npes = ghost_adj.recv_npes;
831  // send cell.i_zone and cell.node_id_list
832  std::vector<std::vector<Int>> send_cells;
833  std::vector<MPI_Request> requests;
834  for (auto &[i_part, npes] : send_npes) {
835  auto &send_buf = send_cells.emplace_back();
836  for (auto [m_cell, npe] : npes) {
837  auto [i_zone, i_sect, i_cell, n_node] = m_to_cell_index_[m_cell];
838  assert(npe == n_node);
839  auto &conn = connectivities_.at(i_zone).at(i_sect);
840  auto &index = conn.index;
841  auto &nodes = conn.nodes;
842  auto *i_node_list = &(nodes[index[i_cell]]);
843  send_buf.emplace_back(i_zone);
844  for (int i = 0; i < npe; ++i) {
845  send_buf.emplace_back(i_node_list[i]);
846  }
847  }
848  int n_ints = send_buf.size();
849  int tag = i_part;
850  auto &request = requests.emplace_back();
851  MPI_Isend(send_buf.data(), n_ints, kMpiIntType, i_part, tag,
852  MPI_COMM_WORLD, &request);
853  }
854  // recv cell.i_zone and cell.node_id_list
855  std::vector<std::vector<Int>> recv_cells;
856  for (auto &[i_part, npes] : recv_npes) {
857  auto &recv_buf = recv_cells.emplace_back();
858  int n_ints = 0;
859  for (auto [m_cell, npe] : npes) {
860  ++n_ints;
861  n_ints += npe;
862  }
863  int tag = rank_;
864  recv_buf.resize(n_ints);
865  auto &request = requests.emplace_back();
866  MPI_Irecv(recv_buf.data(), n_ints, kMpiIntType, i_part, tag,
867  MPI_COMM_WORLD, &request);
868  }
869  // wait until all send/recv finish
870  std::vector<MPI_Status> statuses(requests.size());
871  MPI_Waitall(requests.size(), requests.data(), statuses.data());
872  requests.clear();
873  return recv_cells;
874  }

935  auto &holder_info = m_to_cell_index_[m_holder];
936  auto &sharer_info = m_to_cell_index_[m_sharer];
937  auto i_zone = holder_info.i_zone;
938  // find the common nodes of the holder and the sharer
939  auto i_node_cnt = std::unordered_map<Int, Int>();
940  auto &conn_i_zone = connectivities_.at(i_zone);
941  auto &holder_conn = conn_i_zone.at(holder_info.i_sect);
942  auto &sharer_conn = conn_i_zone.at(sharer_info.i_sect);
943  auto &holder_nodes = holder_conn.nodes;
944  auto &sharer_nodes = sharer_conn.nodes;
945  auto holder_head = holder_conn.index[holder_info.i_cell];
946  auto sharer_head = sharer_conn.index[sharer_info.i_cell];
947  for (int i = 0; i < holder_info.n_node; ++i)
948  ++i_node_cnt[holder_nodes[holder_head + i]];
949  for (int i = 0; i < sharer_info.n_node; ++i)
950  ++i_node_cnt[sharer_nodes[sharer_head + i]];
951  auto common_nodes = std::vector<Int>();
952  common_nodes.reserve(9/* at most 9 nodes on a Face */);
953  for (auto [i_node, cnt] : i_node_cnt)
954  if (cnt == 2)
955  common_nodes.emplace_back(i_node);
956  int face_npe = common_nodes.size();
957  // let the normal vector point from holder to sharer
958  // see http://cgns.github.io/CGNS_docs_current/sids/conv.figs/hexa_8.png
959  auto &zone = local_cells_[i_zone];
960  auto &holder = zone[holder_info.i_sect][holder_info.i_cell];
961  auto &sharer = zone[sharer_info.i_sect][sharer_info.i_cell];
962  holder.adj_cells_.emplace_back(&sharer);
963  sharer.adj_cells_.emplace_back(&holder);
964  auto *face_node_list = common_nodes.data();
965  coordinate::SortNodesOnFace(holder.coordinate(), &holder_nodes[holder_head],
966  face_node_list, face_npe);
967  auto [coordinate_uptr, integrator_uptr]
968  = BuildIntegratorForFace(face_npe, i_zone, face_node_list);
969  auto face_uptr = std::make_unique<Face>(std::move(coordinate_uptr),
970  std::move(integrator_uptr), &holder, &sharer, local_faces_.size());
971  holder.adj_faces_.emplace_back(face_uptr.get());
972  sharer.adj_faces_.emplace_back(face_uptr.get());
973  local_faces_.emplace_back(std::move(face_uptr));
974  }
