103  void GlobalToNormal(Value* v) const requires(kDimensions == 2) {
104  Vector &p = v->momentum();
105  Scalar p_a = p[X] * a(X) + p[Y] * a(Y);
106  Scalar p_b = p[X] * b(X) + p[Y] * b(Y);
107  p[A] = p_a;
108  p[B] = p_b;
109  }
110  void NormalToGlobal(Value* v) const requires(kDimensions == 2) {
111  Vector &p = v->momentum();
112  Scalar p_x = p[A] * a(X) + p[B] * b(X);
113  Scalar p_y = p[A] * a(Y) + p[B] * b(Y);
114  p[X] = p_x;
115  p[Y] = p_y;
116  }
117  void GlobalToNormal(Value* v) const requires(kDimensions == 3) {
118  Vector &p = v->momentum();
119  Scalar p_a = p.dot(a());
120  Scalar p_b = p.dot(b());
121  Scalar p_c = p.dot(c());
122  p[A] = p_a;
123  p[B] = p_b;
124  p[C] = p_c;
125  }
126  void NormalToGlobal(Value* v) const requires(kDimensions == 3) {
127  Vector &p = v->momentum();
128  Scalar p_x = p[A] * a(X) + p[B] * b(X) + p[C] * c(X);
129  Scalar p_y = p[A] * a(Y) + p[B] * b(Y) + p[C] * c(Y);
130  Scalar p_z = p[A] * a(Z) + p[B] * b(Z) + p[C] * c(Z);
131  p[X] = p_x;
132  p[Y] = p_y;

181  Flux GetFluxOnSubsonicInlet(Conservative const &conservative_interior,
182  Value const &given_value) const {
183  Primitive primitive_interior = Gas::ConservativeToPrimitive(conservative_interior);
184  auto c_interior = Gas::GetSpeedOfSound(primitive_interior);
185  GlobalToNormal(&primitive_interior);
186  auto riemann_interior = -primitive_interior.u()
187  - c_interior * Gas::GammaMinusOneUnderTwo();
188  auto u_cos = given_value[1];
189  auto v_cos = given_value[2];
190  auto w_cos = given_value[3];
191  auto T_total = given_value[4];
192  // TODO(PVC): cache A since it's constant.
193  auto A = a(X) * u_cos + a(Y) * v_cos + a(Z) * w_cos;
194  assert(A < 0.0);
195  auto square_of = [](Scalar x) { return x * x; };
196  auto square_of_A = square_of(A);
197  auto gamma_times_R = Gas::Gamma() * Gas::R();
198  auto larger = [](Scalar a, Scalar b_half, Scalar c) -> Scalar {
199  auto sqrt_delta = std::sqrt(b_half * b_half - a * c);
200  auto x_1 = (-b_half + sqrt_delta) / a;
201  auto x_2 = (-b_half - sqrt_delta) / a;
202  return std::max(x_1, x_2);
203  };
204  auto c_boundary = larger(
205  /* the a in (a x^2 + b x + c = 0) */
206  square_of_A + Gas::GammaMinusOneUnderTwo(),
207  /* 1/2 of the b in (a x^2 + b x + c = 0) */
208  riemann_interior,
209  /* the c in (a x^2 + b x + c = 0) */
210  Gas::GammaMinusOneOverTwo() * square_of(riemann_interior)
211  - square_of_A * gamma_times_R * T_total);
212  auto T_boundary = square_of(c_boundary) / gamma_times_R;
213  auto mach_boundary = Gas::GetMachFromTemperature(T_boundary, T_total);
214  auto p_total = given_value[0];
215  auto p_boundary = Gas::TotalPressureToPressure(mach_boundary, p_total);
216  auto speed_boundary = mach_boundary * c_boundary;
217 #ifdef SOLVE_RIEMANN_PROBLEM_AT_BOUNDARY_
218  Primitive primitive_boundary {
219  /* primitive_boundary.rho() = */p_boundary / Gas::R() / T_boundary,
220  /* primitive_boundary.u() = */u_cos * speed_boundary,
221  /* primitive_boundary.v() = */v_cos * speed_boundary,
222  /* primitive_boundary.w() = */w_cos * speed_boundary,
223  /* primitive_boundary.p() = */p_boundary };
224  GlobalToNormal(&primitive_boundary);
225  return NormalPrimitivePairToGlobalFlux(
226  primitive_interior, primitive_boundary);
227 #else
228  auto &primitive_boundary = primitive_interior; // reuse the memory
229  primitive_boundary.p() = p_boundary;
230  primitive_boundary.rho() = p_boundary / Gas::R() / T_boundary;
231  primitive_boundary.u() = u_cos * speed_boundary;
232  primitive_boundary.v() = v_cos * speed_boundary;
233  primitive_boundary.w() = w_cos * speed_boundary;
234  return GlobalPrimitiveToGlobalFlux(&primitive_boundary);
235 #endif
236  }
237  Flux GetFluxOnSubsonicInletOld(
238  Conservative const &conservative_interior,
239  Conservative const &conservative_exterior) const {
240  auto primitive_interior = Gas::ConservativeToPrimitive(conservative_interior);
241  auto primitive_exterior = Gas::ConservativeToPrimitive(conservative_exterior);
242  Primitive primitive_boundary = primitive_exterior;
243  Scalar u_normal_exterior = primitive_exterior.velocity().dot(normal());
244  Scalar u_normal_interior = primitive_interior.velocity().dot(normal());
245  Scalar u_normal_jump = u_normal_exterior - u_normal_interior;
246  Scalar c_interior = Gas::GetSpeedOfSound(primitive_interior);
247  Scalar rho_c_interior = primitive_interior.rho() * (u_normal_exterior > 0 ? c_interior : -c_interior);
248  primitive_boundary.p() = (primitive_interior.p() + primitive_exterior.p()
249  + rho_c_interior * u_normal_jump) * 0.5;
250  Scalar p_jump = primitive_exterior.p() - primitive_boundary.p();
251  primitive_boundary.rho() -= p_jump / (c_interior * c_interior);
252  primitive_boundary.velocity() += (p_jump / rho_c_interior) * normal();
253  return GlobalPrimitiveToGlobalFlux(&primitive_boundary);
254  }

282  Conservative const &conservative_interior,
283  Conservative const &conservative_exterior) const {
284  auto primitive_interior = Gas::ConservativeToPrimitive(conservative_interior);
285  auto primitive_exterior = Gas::ConservativeToPrimitive(conservative_exterior);
286  Primitive primitive_boundary = primitive_interior;

288  Scalar p_jump = primitive_interior.p() - primitive_boundary.p();

293  primitive_boundary.velocity() += (p_jump / rho_c_interior) * normal();

296  Flux GetFluxOnSmartBoundary(
297  Conservative const &conservative_interior,
298  Conservative const &conservative_exterior) const {
299  return GetFluxUpwind(conservative_interior, conservative_exterior);
300  auto primitive = Gas::ConservativeToPrimitive(conservative_interior);
301  Scalar a = Gas::GetSpeedOfSound(primitive);
302  Scalar u_normal = primitive.velocity().dot(normal());
303  Flux flux;
304  if (u_normal < 0) { // inlet
305  if (u_normal + a < 0) {
306  flux = GetFluxOnSupersonicInlet(conservative_exterior);
307  } else {
308  flux = GetFluxOnSubsonicInlet(conservative_interior, conservative_exterior);
309  }
310  } else { // outlet
311  if (u_normal - a > 0) {
312  flux = GetFluxOnSupersonicOutlet(conservative_interior);
313  } else {
314  flux = GetFluxOnSubsonicOutlet(conservative_interior, conservative_exterior);
315  }
316  }
317  return flux;
318  }
