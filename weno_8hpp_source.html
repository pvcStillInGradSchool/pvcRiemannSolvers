25 template <std::floating_point Scalar, int kComponents, int kDegrees>
26 class Smoothness {
27  public:
28  using Taylor = basis::Taylor<Scalar, 3, kDegrees>;
29  using Index = typename Taylor::Index;
30  static constexpr int K = kComponents;
31  static constexpr int N = Taylor::N;
32  static constexpr int D = Taylor::D;
33  static constexpr int P = Taylor::P;
34  using MatKx1 = algebra::Matrix<Scalar, K, 1>;
35  using MatKxN = algebra::Matrix<Scalar, K, N>;

37  static void AddSmoothness(const MatKxN &integral, Scalar weight,
38  int i_pdv_begin, int i_pdv_end, MatKx1 *total_smoothness) {
39  int i_pdv = i_pdv_begin;
40  MatKx1 smoothness = integral.col(i_pdv);
41  while (++i_pdv < i_pdv_end) {
42  smoothness += integral.col(i_pdv);
43  }
44  smoothness *= weight;
45  *total_smoothness += smoothness;
46  }

55  static Scalar GetWeight(Scalar volume, int k) {
56  auto k_factorial = std::tgamma(k + 1);
57  return std::pow(volume, k * 2. / D - 1) / (k_factorial * k_factorial);
58  }
59 
60  static MatKx1 GetSmoothness(const MatKxN &integral, Scalar volume) {
61  static_assert(P > 0);
62  MatKx1 smoothness = integral.col(Index::X);
63  smoothness += integral.col(Index::Y);
64  smoothness += integral.col(Index::Z);
65  smoothness *= GetWeight(volume, 1);
66  static_assert(Taylor::CountBasis(0) == Index::X);
67  static_assert(Taylor::CountBasis(1) == Index::XX);
68  static_assert(Taylor::CountBasis(2) == Index::XXX);
69  static_assert(Taylor::CountBasis(3) == Index::XXXX);
70  for (int p = 2; p <= P; ++p) {
71  AddSmoothness(integral, GetWeight(volume, p),
72  Taylor::CountBasis(p - 1), Taylor::CountBasis(p), &smoothness);
73  }
74  return smoothness;
75  }
76 };

78 template <mini::polynomial::Modal Projection>
79 auto GetSmoothness(const Projection &proj) {
80  using Coeff = typename Projection::Coeff;
81  using Global = typename Projection::Global;
82  using Taylor = typename Projection::Taylor;
83  using Scalar = typename Projection::Scalar;
84  constexpr int K = Projection::K;
85  constexpr int P = Projection::P;
86  auto mat_pdv_func = [&proj](Global const &xyz) {
87  auto local = xyz; local -= proj.center();
88  auto mat_pdv = Taylor::GetPartialDerivatives(local, proj.GetCoeffOnTaylorBasis());
89  mat_pdv = mat_pdv.cwiseProduct(mat_pdv);
90  return mat_pdv;
91  };
92  auto integral = integrator::Integrate(mat_pdv_func, proj.integrator());
93  auto volume = proj.basis().Measure();
94  return Smoothness<Scalar, K, P>::GetSmoothness(integral, volume);
95 }

141  auto const &my_projection = my_cell_->polynomial().projection();
142  auto my_average = my_projection.average();

154  std::cout << std::scientific << std::setprecision(3) <<
155  GetSmoothness(adj_proj).transpose();
156  }
157  }
158  old_projections_.emplace_back(my_projection);
159  if (verbose_) {

161  std::cout << std::scientific << std::setprecision(3) <<
162  GetSmoothness(old_projections_.back()).transpose();
163  }

165  assert(&(new_projection_ptr_->basis()) == &(my_projection.basis()));
166  }
167  void Reconstruct() {
168  int adj_cnt = my_cell_->adj_cells_.size();
169  // initialize weights
170  auto weights = std::vector<Value>(adj_cnt + 1, weights_);
171  weights.back() *= -adj_cnt;
172  weights.back().array() += 1.0;
173  // modify weights by smoothness
174  for (int i = 0; i <= adj_cnt; ++i) {
175  auto beta = GetSmoothness(old_projections_[i]);
176  beta.array() += eps_;
177  beta.array() *= beta.array();
178  weights[i].array() /= beta.array();
179  }
180  // normalize these weights
181  Value sum; sum.setZero();
182  sum = std::accumulate(weights.begin(), weights.end(), sum);
183  assert(weights.size() == adj_cnt + 1);
184  for (auto &weight : weights) {
185  weight.array() /= sum.array();
186  }
187  // build the new (weighted) projection
188  auto &new_projection = old_projections_.back();
189  new_projection *= weights.back();
190  for (int i = 0; i < adj_cnt; ++i) {
191  old_projections_[i] *= weights[i];
192  new_projection += old_projections_[i];
193  }
194  }

230  static bool IsNotSmooth(const Cell &cell) {
231  constexpr int components[] = { 0, Cell::K-1 };
232  auto max_abs_averages = cell.polynomial().average();
233  for (int i : components) {
234  max_abs_averages[i] = std::max(1e-9, std::abs(max_abs_averages[i]));
235  }
236  typename Cell::Value sum_abs_differences; sum_abs_differences.setZero();
237  auto my_values = cell.GlobalToValue(cell.center());
238  for (const Cell *adj_cell : cell.adj_cells_) {
239  auto adj_values = adj_cell->GlobalToValue(cell.center());
240  auto adj_averages = adj_cell->polynomial().average();
241  for (int i : components) {
242  sum_abs_differences[i] += std::abs(my_values[i] - adj_values[i]);
243  max_abs_averages[i] = std::max(max_abs_averages[i],
244  std::abs(adj_averages[i]));
245  }
246  }
247  constexpr auto volume_power = (Cell::P + 1.0) / 2.0 / Cell::D;
248  auto divisor = std::pow(cell.volume(), volume_power);
249  divisor *= cell.adj_cells_.size();
250  constexpr auto smoothness_reference = Cell::P < 3 ? 1.0 : 3.0;
251  for (int i : components) {
252  auto smoothness = sum_abs_differences[i] / max_abs_averages[i] / divisor;
253  if (smoothness > smoothness_reference) {
254  return true; // if any component is not smooth
255  }
256  }
257  return false; // if all components are smooth
258  }

290  void ReconstructOnFace(const Face &adj_face) {
291  assert(my_cell_->adj_faces_.size() == my_cell_->adj_cells_.size());
292  int adj_cnt = my_cell_->adj_faces_.size();
293  // build eigen-matrices in the rotated coordinate system
294  const auto &big_u = my_cell_->polynomial().average();
295  // TODO(PVC): take average of eigen matrices on all quadrature points
296  const auto &const_riemann = face_to_riemanns_(adj_face).at(0);
297  auto *riemann = const_cast<Riemann *>(&const_riemann);
298  riemann->UpdateEigenMatrices(big_u);
299  // initialize weights
300  auto weights = std::vector<Value>(adj_cnt + 1, weights_);
301  weights.back() *= -adj_cnt;
302  weights.back().array() += 1.0;
303  // modify weights by smoothness
304  auto rotated_projections = old_projections_;
305  for (int i = 0; i <= adj_cnt; ++i) {
306  auto &projection_i = rotated_projections[i];
307  projection_i.LeftMultiply(riemann->L());
308  auto beta = GetSmoothness(projection_i);
309  beta.array() += eps_;
310  beta.array() *= beta.array();
311  weights[i].array() /= beta.array();
312  }
313  // normalize these weights
314  Value sum; sum.setZero();
315  sum = std::accumulate(weights.begin(), weights.end(), sum);
316  assert(weights.size() == adj_cnt + 1);
317  for (auto &weight : weights) {
318  weight.array() /= sum.array();
319  }
320  // build the new (weighted) projection
321  auto &new_projection = rotated_projections.back();
322  new_projection *= weights.back();
323  for (int i = 0; i < adj_cnt; ++i) {
324  rotated_projections[i] *= weights[i];
325  new_projection += rotated_projections[i];
326  }
327  // rotate the new projection back to the global system
328  new_projection.LeftMultiply(riemann->R());
329  // scale the new projection by volume
330  auto adj_volume = adj_face.other(my_cell_)->volume();
331  new_projection *= adj_volume;
332  new_projection_ += new_projection;
333  total_volume_ += adj_volume;
334  }

345  ReconstructOnFace(*adj_face);
346  }
347  new_projection_ /= total_volume_;
