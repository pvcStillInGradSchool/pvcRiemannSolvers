226  const std::vector<Int> &cost_of_each_vertex = {},

233  "`Int` and `idx_t` must have the same size.");

313 template <std::integral Int>
314 std::vector<Int> GetNodeParts(
315  const metis::Mesh<Int>& mesh, const std::vector<Int>& cell_parts,
316  Int n_parts) {
317  Int n_nodes = mesh.CountNodes();
318  Int n_cells = mesh.CountCells();
319  auto node_parts = std::vector<Int>(n_nodes, n_parts);
320  auto node_pointer = &(mesh.nodes(0));
321  auto curr_range_pointer = &(mesh.range(0));
322  for (Int i = 0; i < n_cells; ++i) {
323  auto part_value = cell_parts[i];
324  auto head = *curr_range_pointer++;
325  auto tail = *curr_range_pointer;
326  for (Int j = head; j < tail; ++j) {
327  Int node_index = *node_pointer++;
328  /* If more than one parts share a common node, then the
329  node belongs to the min-id part. */
330  if (part_value < node_parts[node_index])
331  node_parts[node_index] = part_value;
332  }
333  }
334  return node_parts;
335 }
