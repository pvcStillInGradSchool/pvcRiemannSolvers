85  template <std::ranges::input_range Range, class FaceToCell>
86  void CacheCorrectionGradients(Range &&faces, FaceToCell &&face_to_cell,
87  std::vector<FaceCache> *cache) {
88  for (const Face &face : faces) {
89  assert(cache->size() == face.id());
90  auto &curr_face = cache->emplace_back();
91  const auto &face_integrator = face.integrator();
92  const auto &cell = face_to_cell(face);
93  const auto &cell_integrator = cell.integrator();
94  const auto &cell_polynomial = cell.polynomial();
95  const auto &cell_basis = cell_polynomial.interpolation().basis();
96  int i_face = cell_polynomial.FindFaceId(face.coordinate().center());
97  assert(kFaceQ == face.integrator().CountPoints());
98  for (int f = 0; f < kFaceQ; ++f) {
99  Global const &face_normal = face_integrator.GetNormalFrame(f)[0];
100  auto &[curr_line, flux_point] = curr_face.at(f);
101  auto &flux_point_coord = face_integrator.GetGlobal(f);
102  auto [i, j, k] = cell_polynomial.FindCollinearIndex(flux_point_coord, i_face);
103  switch (i_face) {
104  case 0:
105  assert(k == -1);
106  flux_point.ijk = cell_basis.index(i, j, 0);
107  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
108  flux_point.normal =
109  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Z);
110  flux_point.scale = -flux_point.normal.norm();
111  flux_point.xyz = Z;
112  assert(Collinear(face_normal, flux_point.normal));
113  for (k = 0; k < kLineQ; ++k) {
114  auto ijk = cell_basis.index(i, j, k);
115  auto &local = cell_integrator.GetLocal(ijk);
116  auto g_prime = vincent_.LocalToLeftDerivative(local[Z]);
117  curr_line[k].g_prime = g_prime;
118  curr_line[k].ijk = ijk;
119  }
120  break;
121  case 1:
122  assert(j == -1);
123  flux_point.ijk = cell_basis.index(i, 0, k);
124  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
125  flux_point.normal =
126  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Y);
127  flux_point.scale = -flux_point.normal.norm();
128  flux_point.xyz = Y;
129  assert(Collinear(face_normal, flux_point.normal));
130  for (j = 0; j < kLineQ; ++j) {
131  auto ijk = cell_basis.index(i, j, k);
132  auto &local = cell_integrator.GetLocal(ijk);
133  auto g_prime = vincent_.LocalToLeftDerivative(local[Y]);
134  curr_line[j].g_prime = g_prime;
135  curr_line[j].ijk = ijk;
136  }
137  break;
138  case 2:
139  assert(i == -1);
140  flux_point.ijk = cell_basis.index(kLineQ - 1, j, k);
141  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
142  flux_point.normal =
143  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(X);
144  flux_point.scale = +flux_point.normal.norm();
145  flux_point.xyz = X;
146  assert(Collinear(face_normal, flux_point.normal));
147  for (i = 0; i < kLineQ; ++i) {
148  auto ijk = cell_basis.index(i, j, k);
149  auto &local = cell_integrator.GetLocal(ijk);
150  auto g_prime = vincent_.LocalToRightDerivative(local[X]);
151  curr_line[i].g_prime = g_prime;
152  curr_line[i].ijk = ijk;
153  }
154  break;
155  case 3:
156  assert(j == -1);
157  flux_point.ijk = cell_basis.index(i, kLineQ - 1, k);
158  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
159  flux_point.normal =
160  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Y);
161  flux_point.scale = +flux_point.normal.norm();
162  flux_point.xyz = Y;
163  assert(Collinear(face_normal, flux_point.normal));
164  for (j = 0; j < kLineQ; ++j) {
165  auto ijk = cell_basis.index(i, j, k);
166  auto &local = cell_integrator.GetLocal(ijk);
167  auto g_prime = vincent_.LocalToRightDerivative(local[Y]);
168  curr_line[j].g_prime = g_prime;
169  curr_line[j].ijk = ijk;
170  }
171  break;
172  case 4:
173  assert(i == -1);
174  flux_point.ijk = cell_basis.index(0, j, k);
175  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
176  flux_point.normal =
177  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(X);
178  flux_point.scale = -flux_point.normal.norm();
179  flux_point.xyz = X;
180  assert(Collinear(face_normal, flux_point.normal));
181  for (i = 0; i < kLineQ; ++i) {
182  auto ijk = cell_basis.index(i, j, k);
183  auto &local = cell_integrator.GetLocal(ijk);
184  auto g_prime = vincent_.LocalToLeftDerivative(local[X]);
185  curr_line[i].g_prime = g_prime;
186  curr_line[i].ijk = ijk;
187  }
188  break;
189  case 5:
190  assert(k == -1);
191  flux_point.ijk = cell_basis.index(i, j, kLineQ - 1);
192  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
193  flux_point.normal =
194  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Z);
195  flux_point.scale = +flux_point.normal.norm();
196  flux_point.xyz = Z;
197  assert(Collinear(face_normal, flux_point.normal));
198  for (k = 0; k < kLineQ; ++k) {
199  auto ijk = cell_basis.index(i, j, k);
200  auto &local = cell_integrator.GetLocal(ijk);
201  auto g_prime = vincent_.LocalToRightDerivative(local[Z]);
202  curr_line[k].g_prime = g_prime;
203  curr_line[k].ijk = ijk;
204  }
205  break;
206  default:
207  assert(false);
208  }
209  }
210  }

250  Column residual = this->Base::GetResidualColumn();
251  if (Polynomial::kLocal) {

270  protected: // override virtual methods defined in Base
271  void AddFluxDivergence(Cell const &cell, Scalar *residual) const override {
272  assert(residual);
273  const auto &integrator = cell.integrator();
274  const auto &polynomial = cell.polynomial();
275  auto &flux = const_cast<General *>(this)->flux_matrices_.at(cell.id());
276  for (int q = 0, n = integrator.CountPoints(); q < n; ++q) {
277  FluxMatrix global_flux = Base::GetFluxMatrix(cell, q);
278  flux[q] = polynomial.GlobalFluxToLocalFlux(global_flux, q);
279  }
280  for (int q = 0, n = integrator.CountPoints(); q < n; ++q) {
281  auto const &grad = polynomial.GetBasisLocalGradients(q);
282  Value value = flux[0] * grad.col(0);
283  for (int k = 1; k < n; ++k) {
284  value += flux[k] * grad.col(k);
285  }
286  Polynomial::MinusValue(value, residual, q);
287  }
288  }

296  std::pair<Value, Value> GetFluxForHolderAndSharer(const Riemann &riemann,
297  const Cell &holder, FluxPointCache const &holder_cache,
298  const Cell &sharer, FluxPointCache const &sharer_cache,
299  bool use_cached_flux_on_sharer) const
300  requires(!mini::riemann::Diffusive<Riemann>) {
301  Value u_holder = holder.polynomial().GetValue(holder_cache.ijk);
302  Value u_sharer = sharer.polynomial().GetValue(sharer_cache.ijk);
303  Value f_upwind = riemann.GetFluxUpwind(u_holder, u_sharer);
304  assert(Collinear(holder_cache.normal, sharer_cache.normal));
305  Value f_holder = f_upwind * holder_cache.scale;
306  MinusCachedFlux(&f_holder, holder.id(), holder_cache);
307  Value f_sharer = f_upwind * (-sharer_cache.scale);
308  if (use_cached_flux_on_sharer) {
309  MinusCachedFlux(&f_sharer, sharer.id(), sharer_cache);
310  } else {
311  auto f_mat_sharer = Riemann::Convection::GetFluxMatrix(u_sharer);
312  f_sharer -= f_mat_sharer * sharer_cache.normal;
313  }
314  return { f_holder, f_sharer };
315  }
316  std::pair<Value, Value> GetFluxForHolderAndSharer(const Riemann &riemann,
317  const Cell &holder, FluxPointCache const &holder_cache,
318  const Cell &sharer, FluxPointCache const &sharer_cache,
319  bool use_cached_flux_on_sharer) const
320  requires(mini::riemann::ConvectiveDiffusive<Riemann>) {
321  auto [u_holder, du_holder, ddu_holder] =
322  holder.polynomial().GetGlobalValueGradientHessian(holder_cache.ijk);
323  auto [u_sharer, du_sharer, ddu_sharer] =
324  sharer.polynomial().GetGlobalValueGradientHessian(sharer_cache.ijk);
325  Value f_upwind = riemann.GetFluxUpwind(u_holder, u_sharer);
326  assert(Collinear(holder_cache.normal, sharer_cache.normal));
327  const auto &normal = riemann.normal();
328  auto du_common = riemann.GetCommonGradient(normal,
329  u_holder, u_sharer, du_holder, du_sharer, ddu_holder, ddu_sharer);
330  Value u_common = (u_holder + u_sharer) / 2;
331  auto property = Riemann::Diffusion::GetPropertyOnCell(
332  holder.id(), holder_cache.ijk);
333  property += Riemann::Diffusion::GetPropertyOnCell(
334  sharer.id(), sharer_cache.ijk);
335  property *= 0.5;
336  Riemann::MinusViscousFlux(&f_upwind, property, u_common, du_common, normal);
337  Value f_holder = f_upwind * holder_cache.scale;
338  MinusCachedFlux(&f_holder, holder.id(), holder_cache);
339  Value f_sharer = f_upwind * (-sharer_cache.scale);
340  if (use_cached_flux_on_sharer) {
341  MinusCachedFlux(&f_sharer, sharer.id(), sharer_cache);
342  } else {
343  auto f_mat_sharer = Riemann::Convection::GetFluxMatrix(u_sharer);
344  const auto &property = Riemann::Diffusion::GetPropertyOnCell(
345  sharer.id(), sharer_cache.ijk);
346  Riemann::MinusViscousFlux(&f_mat_sharer, property, u_sharer, du_sharer);
347  f_sharer -= f_mat_sharer * sharer_cache.normal;
348  }
349  return { f_holder, f_sharer };
350  }

409  void AddFluxToHolderAndSharer(Face const &face,
410  Scalar *holder_data, Scalar *sharer_data) const override {
411  const auto &riemanns = this->GetRiemannSolvers(face);
412  const auto &holder = face.holder();
413  const auto &sharer = face.sharer();
414  auto const &holder_cache = holder_cache_[face.id()];
415  auto const &sharer_cache = sharer_cache_[face.id()];
416  assert(kFaceQ == face.integrator().CountPoints());
417  for (int f = 0; f < kFaceQ; ++f) {
418  auto &[holder_solution_points, holder_flux_point] = holder_cache[f];
419  auto &[sharer_solution_points, sharer_flux_point] = sharer_cache[f];
420  auto [f_holder, f_sharer] = GetFluxForHolderAndSharer(riemanns[f],
421  holder, holder_flux_point,
422  sharer, sharer_flux_point, sharer_data);
423  assert(holder_data);
424  for (auto [g_prime, ijk] : holder_solution_points) {
425  Value f_correction = f_holder * g_prime;
426  Polynomial::MinusValue(f_correction, holder_data, ijk);
427  }
428  if (nullptr == sharer_data) { continue; }
429  for (auto [g_prime, ijk] : sharer_solution_points) {
430  Value f_correction = f_sharer * g_prime;
431  Polynomial::MinusValue(f_correction, sharer_data, ijk);
432  }
433  }
434  }
