54 class Huynh(Polynomial):
55  r"""The left and right \f$ g(\xi) \f$ in Huynh's FR schemes.
56 
57  The right \f$ g(\xi) \f$ with lumping conditions up to the \f$ m \f$th derivative for a \f$ k \f$-degree \f$ u^h \f$ is a \f$ (k+1) \f$-degree polynomial, which satisfies
58  (a) \f$ (m+1) \f$ lumping conditions \f$ g^{(0)}(-1) = \dots = g^{(m)}(-1) = 0 \f$, and
59  (b) \f$ (k-m) \f$ orthogonality conditions \f$ g \perp \mathbb{P}_{0} \land \cdots \land g \perp \mathbb{P}_{k-m-1} \f$, and
60  (c) \f$ 1 \f$ boundary value condition \f$ g(1) = 1 \f$.
61  """

71  # lumping conditions
72  factorials = np.ones(n_term)
73  for i in range(2, n_term):
74  factorials[i] = factorials[i - 1] * i
75  for l in range(n_lump):
76  for i in range(n_term):
77  if i >= l:
78  a[l][i] = (-1)**(i - l) * factorials[i] / factorials[i - l]
79  # orthogonality conditions
