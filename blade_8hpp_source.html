132  Section GetSection(Scalar y_ratio) const {
133  assert(0 <= y_ratio && y_ratio <= 1);
134  auto y_value = y_ratio * GetSpan();
135  auto itr = std::lower_bound(y_values_.begin(), y_values_.end(), y_value);
136  assert(itr != y_values_.end());
137  auto i_1 = itr - y_values_.begin();
138  auto chord = chords_[i_1];
139  auto twist = twists_[i_1];
140  auto a_1 = airfoils_[i_1];
141  auto a_0 = a_1;
142  auto w_1 = 1.0, w_0 = 0.0;
143  if (*itr != y_value) {
144  // linear interpolation
145  assert(i_1 > 0);
146  auto i_0 = i_1 - 1;
147  auto lambda_1 = (itr[0] - y_value) / (itr[0] - itr[-1]);
148  auto lambda_0 = 1 - lambda_1;
149  chord = chord * lambda_1 + chords_[i_0] * lambda_0;
150  twist = twist * lambda_1 + twists_[i_0] * lambda_0;
151  a_0 = airfoils_[i_0];
152  w_0 = lambda_0;
153  w_1 = lambda_1;
154  }
155  return Section(*this, y_ratio, chord, twist, a_0, w_0, a_1, w_1);
