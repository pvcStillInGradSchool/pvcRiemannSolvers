45  static std::pair<Scalar, Scalar>
46  Intersect(const Cell &cell, const Blade &blade) {
47  const Global &p = blade.P();
48  const Global &q = blade.Q();
49  const Global &pq = blade.PQ();
50  Scalar r_ratio{-1}, s_ratio{-1};
51  for (const Face *face : cell.adj_faces_) {
52  if (Valid(r_ratio) && Valid(s_ratio)) {
53  break;
54  }
55  const auto &integrator = face->integrator();
56  // Currently, only triangle is supported.
57  assert(integrator.coordinate().CountCorners() == 3);
58  Global pa = integrator.coordinate().GetGlobal(0) - p;
59  Global pb = integrator.coordinate().GetGlobal(1) - p;
60  Global pc = integrator.coordinate().GetGlobal(2) - p;
61  Scalar ratio = -1.0;
62  mini::geometry::Intersect(pa, pb, pc, pq, &ratio);
63  if (Valid(ratio)) {
64  if (!Valid(r_ratio)) {
65  r_ratio = ratio;
66  } else if (!Valid(s_ratio)) {
67  s_ratio = ratio;
68  } else {
69  // More than two common points are found.
70  assert(false);
71  }
72  }
73  }
74  if (Valid(r_ratio) && !Valid(s_ratio)) {
75  // If only one common point is found (R is always found before S),
76  // then either P or Q is inside.
77  s_ratio = r_ratio < 0.5 ? 0 : 1; // p_ratio = 0, q_ratio = 1
78  }
79  if (Valid(r_ratio) && Valid(s_ratio)) {
80  if (r_ratio > s_ratio) {
81  std::swap(r_ratio, s_ratio);
82  }
83  }
84  return {r_ratio, s_ratio};
85  }

87  static std::pair<Force, Scalar>
88  GetSourceValue(const Cell &cell, const Section &section, const Global &xyz) {
89  auto value = cell.GlobalToValue(xyz);
90  auto &cv = *reinterpret_cast<Conservative *>(&value);
91  auto uvw = cv.momentum() / cv.mass();
92  Force force = section.GetForce(cv.mass(), uvw);
93  auto power = force.transpose() * uvw;
94  return {force, power};
95  }

97  void UpdateCoeff(const Cell &cell, const Blade &blade, Scalar *coeff_data) {
98  auto [r_ratio, s_ratio] = Intersect(cell, blade);
99  if (r_ratio < s_ratio) {
100  // r_ratio is always set before s_ratio
101  assert(Valid(r_ratio) && Valid(s_ratio));
102  // Integrate along RS:
103  auto line = mini::integrator::Line<Scalar, 1, 4>(r_ratio, s_ratio);
104  auto integrand = [&cell, &blade](Scalar ratio){
105  auto section = blade.GetSection(ratio);
106  auto xyz = section.GetOrigin();
107  auto [force, power] = GetSourceValue(cell, section, xyz);
108  using Mat1xN = mini::algebra::Matrix<Scalar, 1, Cell::N>;
109  Mat1xN basis_values = cell.GlobalToBasisValues(xyz);
110  using Mat4xN = mini::algebra::Matrix<Scalar, 4, Cell::N>;
111  Mat4xN product;
112  product.row(0) = force[0] * basis_values;
113  product.row(1) = force[1] * basis_values;
114  product.row(2) = force[2] * basis_values;
115  product.row(3) = power * basis_values;
116  return product;
117  };
118  auto integral = mini::integrator::Integrate(integrand, line);
119  integral *= blade.GetSpan();
120  auto *coeff = reinterpret_cast<Coeff *>(coeff_data);
121  coeff->row(1) += integral.row(0);
122  coeff->row(2) += integral.row(1);
123  coeff->row(3) += integral.row(2);
124  coeff->row(4) += integral.row(3);
125  }
126  }

138  void GetForces(const Cell &cell, double t_curr, std::vector<Global> *forces,
139  std::vector<Global> *points, std::vector<Scalar> *weights) {
140  for (auto &rotor : rotors_) {
141  rotor.UpdateAzimuth(t_curr);
142  for (int i = 0, n = rotor.CountBlades(); i < n; ++i) {
143  const Blade &blade = rotor.GetBlade(i);
144  auto [r_ratio, s_ratio] = Intersect(cell, blade);
145  if (r_ratio >= s_ratio) {
146  continue;
147  }
148  constexpr int Q = 4;
149  auto line = mini::integrator::Line<Scalar, 1, Q>(r_ratio, s_ratio);
150  for (int q = 0; q < Q; ++q) {
151  auto ratio = line.GetGlobal(q);
152  auto section = blade.GetSection(ratio);
153  auto xyz = section.GetOrigin();
154  auto [force, power] = GetSourceValue(cell, section, xyz);
155  forces->emplace_back(force);
156  points->emplace_back(xyz);
157  weights->emplace_back(line.GetGlobalWeight(q) * blade.GetSpan());
158  }
159  }
160  }
