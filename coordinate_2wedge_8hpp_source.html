27  using Base = Cell<Scalar>;
28  using typename Base::Real;
29  using typename Base::Local;

41  void BuildCenter() final {
42  Scalar a = 1.0 / 3;
43  center_ = this->LocalToGlobal(a, a, 0);
44  }
45  const Global &center() const final {
46  return center_;
47  }

49  protected:
50  static int GetTriangleId(const size_t *cell_nodes, size_t *face_nodes,
51  int face_n_node/* number of nodes on triangle */) {
52  int i_face = 0;
53  for (int f = 0; f < face_n_node; ++f) {
54  if (cell_nodes[4] == face_nodes[f]) {
55  i_face = 1;
56  break;
57  }
58  }
59  return i_face;
60  }
61 
62  static int GetQuadrangleId(const size_t *cell_nodes, size_t *face_nodes,
63  int face_n_node/* number of nodes on quadragle */) {
64  int c_sum = 0; // sum of c's in (0, 1, 2)
65  for (int f = 0; f < face_n_node; ++f) {
66  for (int c = 0; c < 3; ++c) {
67  if (cell_nodes[c] == face_nodes[f]) {
68  c_sum += c;
69  break;
70  }
71  }
72  }
73  int i_face;
74  switch (c_sum) {
75  case 1: i_face = 0; break;
76  case 2: i_face = 2; break;
77  case 3: i_face = 1; break;
78  default: assert(false);
79  }
80  return i_face;
81  }
82 };

296  Scalar z_local, Local *grads) {
297  auto c_local = 1.0 - a_local - b_local;
298  auto factor_a = 4 * a_local;
299  auto factor_b = 4 * b_local;
300  auto factor_c = 4 * c_local;
301  auto factor_ab = factor_a * b_local;
302  auto factor_bc = factor_b * c_local;
303  auto factor_ca = factor_c * a_local;
304  // local_coords_[i][Z] = -1
305  auto factor_z = (1 - z_local) / 2;
306  auto grad_z = -0.5;
307  // shape[6] = 4 * a * b * factor_z
308  grads[6][A] = factor_b * factor_z;
309  grads[6][B] = factor_a * factor_z;
310  grads[6][Z] = factor_ab * grad_z;
311  // shape[7] = 4 * b * c * factor_z
312  grads[7][A] = -factor_b * factor_z;
313  grads[7][B] = (factor_c - factor_b) * factor_z;
314  grads[7][Z] = factor_bc * grad_z;
315  // shape[8] = 4 * c * a * factor_z
316  grads[8][A] = (factor_c - factor_a) * factor_z;
317  grads[8][B] = -factor_a * factor_z;
318  grads[8][Z] = factor_ca * grad_z;
319  // local_coords_[i][Z] = 0
320  factor_z = (1 - z_local * z_local);
321  grad_z = -2 * z_local;
322  // shape[9] = a_local * factor_z;
323  grads[9][A] = factor_z;
324  grads[9][B] = 0;
325  grads[9][Z] = a_local * grad_z;
326  // shape[10] = b_local * factor_z;
327  grads[10][A] = 0;
328  grads[10][B] = factor_z;
329  grads[10][Z] = b_local * grad_z;

331  grads[11][A] = -factor_z;
332  grads[11][B] = -factor_z;
333  grads[11][Z] = c_local * grad_z;

335  factor_z = (1 + z_local) / 2;
336  grad_z = +0.5;
337  // shape[12] = 4 * a * b * factor_z

339  grads[12][B] = factor_a * factor_z;
