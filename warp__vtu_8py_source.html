55  """Warp a single VTU file by GPU.
56  """
57  grid, n_point, vtk_points_data, data_array = read_vtu(filename, component)
58  # read data for shifting
59  pairs_on_host = np.ndarray((n_point, 2))
60  for p in range(n_point):
61  pairs_on_host[p][0] = vtk_points_data.GetTuple(p)[axis]
62  pairs_on_host[p][1] = data_array.GetTuple(p)[0]
63  # shift coords by GPU
64  module = SourceModule("""
65  __global__ void shift(double *pairs, double scale, int len) {
66  int idx = blockIdx.x * blockDim.x + threadIdx.x;
67  if (idx < len) {
68  double *pair = pairs + idx * 2;
69  pair[0] += scale * pair[1];
70  }
71  }
72  """)
73  func = module.get_function("shift")
74  pairs_on_device = cuda.mem_alloc(pairs_on_host.nbytes)

76  block_size = (1024, 1, 1)
