114  static void MinusViscousFlux(FluxMatrix *flux, Property const &nu,
115  Conservative const &value, Gradient const &gradient) {
116  _MinusViscousFluxByBase(flux, value, gradient);
117  for (int k = 0; k < kComponents; ++k) {

125  _MinusViscousFluxByBase(flux, value, gradient, normal);
126  for (int k = 0; k < kComponents; ++k) {
127  (*flux)[k] -= nu[k] * normal.dot(gradient.col(k));

250  static void UpdateCellResidual(Cell *curr_cell, Scalar *residual_data) {
251  spatial_ptr_->AddFluxDivergence(*curr_cell, residual_data);
252  for (Face *face : curr_cell->adj_faces_) {
253  if (face->holder_ptr() == curr_cell) {
254  spatial_ptr_->AddFluxToHolder(*face, residual_data);
255  } else {
256  spatial_ptr_->AddFluxToSharer(*face, residual_data);
257  }
258 #if !defined(NDEBUG) && defined(ENABLE_SLOW_CONSISTENCY_CHECK)
259 {
260  Coeff dummy;
261  Cell const *other = face->other(curr_cell);
262  assert(other->polynomial().coeff() == Coeff::Zero());
263  Coeff res1, res2;
264  if (face->holder_ptr() == curr_cell) {
265  res1.setZero();
266  res2.setZero();
267  spatial_ptr_->AddFluxToHolder(*face, res1.data());
268  spatial_ptr_->AddFluxToHolderAndSharer(*face, res2.data(), nullptr);
269  assert(res1 == res2);
270  } else {
271  assert(face->holder_ptr() == other);
272  res1.setZero();
273  res2.setZero();
274  spatial_ptr_->AddFluxToSharer(*face, res1.data());
275  spatial_ptr_->AddFluxToHolderAndSharer(*face, dummy.data(), res2.data());
276  assert(res1 == res2);
277  }
278 }
279 #endif
280  }
281  assert(curr_cell->adj_cells_.size() == curr_cell->adj_faces_.size());
282  for (Face *face : curr_cell->boundary_faces_) {
283  assert(face->holder_ptr() == curr_cell);
284  assert(nullptr == face->other(curr_cell));
285  assert(face->HolderToSharer().dot(
286  face->center() - curr_cell->center()) > 0);
287  spatial_ptr_->AddFluxToHolder(*face, residual_data);
288  }
289  assert(curr_cell->boundary_faces_.size() + curr_cell->adj_faces_.size() == 6);
290  }

299  static std::vector<DampingMatrix> BuildDampingMatrices() {
300 #if !defined(NDEBUG) && defined(ENABLE_SLOW_CONSISTENCY_CHECK)
301  std::srand(31415926);
302 #endif
303  auto matrices = std::vector<DampingMatrix>(part().CountLocalCells());
304  for (Cell *curr_cell: part_ptr()->GetLocalCellPointers()) {
305  // Nullify coeffs and properties on all its neighbors:
306  for (Cell *neighbor : curr_cell->adj_cells_) {
307  neighbor->polynomial().SetZero();
308  }
309  // Build the damping matrix column by column:
310  auto &matrix = matrices.at(curr_cell->id());
311  auto GetCellResidual = [](Cell *cell_ptr) -> Coeff {
312  Coeff residual; residual.setZero();
313  SetViscousProperty(cell_ptr, Value::Zero());
314  UpdateCellResidual(cell_ptr, residual.data());
315  residual = -residual;
316  SetViscousProperty(cell_ptr, Value::Ones());
317  UpdateCellResidual(cell_ptr, residual.data());
318  if (!Polynomial::kLocal) {
319  // TODO(PVC): Use the virtual method in Base
320  Scalar *data = residual.data();
321  const auto &integrator = cell_ptr->integrator();
322  for (int q = 0; q < Cell::N; ++q) {
323  auto scale = 1. / integrator.GetJacobianDeterminant(q);
324  data = cell_ptr->polynomial().ScaleValueAt(scale, data);
325  }
326  assert(data == residual.data() + Cell::kFields);
327  }
328  return residual;
329  };
330  auto &curr_polynomial = curr_cell->polynomial();
331  curr_polynomial.SetZero();
332  for (int c = 0; c < Cell::N; ++c) {
333  if (c > 0) {
334  curr_polynomial.SetCoeff(c - 1, Value::Zero());
335  }
336  curr_polynomial.SetCoeff(c, Value::Ones());
337  Coeff residual = GetCellResidual(curr_cell);
338  // Write the residual column into the matrix:
339  matrix.col(c) = residual.row(0);
340 #ifndef NDEBUG
341  for (int r = 1; r < Cell::K; ++r) {
342  if ((residual.row(r) - residual.row(0)).squaredNorm() > 1e-10) {
343  std::cout << residual.row(r) << "\n\n";
344  std::cout << residual.row(0) << "\n\n";
345  assert(false);
346  }
347  }
348 #endif
349  }
350 #if !defined(NDEBUG) && defined(ENABLE_SLOW_CONSISTENCY_CHECK)
351 {
352  Coeff solution = Coeff::Random();
353  curr_cell->polynomial().SetCoeff(solution);
354  assert(solution == curr_cell->polynomial().coeff());
355  Coeff residual = GetCellResidual(curr_cell);
356  assert(solution == curr_cell->polynomial().coeff());
357  for (int k = 0; k < Cell::K; ++k) {
358  auto const &residual_col = residual.row(k).transpose();
359  auto const &solution_col = solution.row(k).transpose();
360  if ((residual_col - matrix * solution_col).norm() > 1e-9) {
361  std::cout << (residual_col).transpose() << "\n\n";
362  std::cout << (matrix * solution_col).transpose() << "\n\n";
363  std::cout << (residual_col - matrix * solution_col).norm() << "\n\n";
364  assert(false);
365  }
366  }
367 }
368 #endif
369  // Scale the damping matrix by Gaussian weights:
370  for (int r = 0; r < Cell::N; ++r) {
371  Scalar scale = curr_cell->integrator().GetLocalWeight(r);
372  Scalar det = curr_cell->integrator().GetJacobianDeterminant(r);
373  if (Polynomial::kLocal) {
374  scale /= det;
375  } else {
376  scale *= det;
377  assert(scale == curr_cell->integrator().GetGlobalWeight(r));
378  }
379  assert(scale > 0);
380  matrix.row(r) *= scale;
381  }
382  }
383  return matrices;
384  }

386  private: // methods for getting the square of jump on a given quadrature point
387  static Value Maximize(Cell const &cell, int i_node) {
388  Value jump_on_curr_node = Value::Zero();
389  Global const &global = cell.integrator().GetGlobal(i_node);
390  Value const &value = cell.polynomial().GetValue(i_node);
391  for (Cell *neighbor : cell.adj_cells_) {
392  Value jump = value - neighbor->polynomial().Extrapolate(global);
393  mini::algebra::Maximize(&jump_on_curr_node, std::pow(jump, 2));
394  }
395  return jump_on_curr_node;
396  }

398  static Value WeightedAverage(Cell const &cell, int i_node) {
399  Value jump_on_curr_node = Value::Zero();
400  Global const &global = cell.integrator().GetGlobal(i_node);
401  Value const &value = cell.polynomial().GetValue(i_node);
402  Scalar weight_sum = 0;
403  for (Cell *neighbor : cell.adj_cells_) {
404  Value jump = value - neighbor->polynomial().Extrapolate(global);
405  // the closer neighbor, the larger weight:
406  Scalar weight = std::exp(-(neighbor->center() - global).squaredNorm());
407  jump_on_curr_node += std::pow(jump, 2) * weight;
408  weight_sum += weight;
409  }
410  return jump_on_curr_node /= weight_sum;
411  }

446  static std::vector<Value> IntegrateJumpOnFaces() {
447  std::vector<Value> jump_integrals;
448  jump_integrals.resize(part().CountLocalCells(), Value::Zero());
449  for (Face const &face : part_ptr()->GetLocalFaces()) {
450  const auto &holder = face.holder();
451  const auto &sharer = face.sharer();
452  const auto &integrator = face.integrator();
453  for (int i = 0; i < integrator.CountPoints(); ++i) {
454  Value jump = spatial_ptr_->GetValueJump(face, i);
455  jump.array() *= jump.array();
456  jump_integrals.at(holder.id()) += jump
457  * (integrator.GetGlobalWeight(i) * face.holder_height());
458  jump_integrals.at(sharer.id()) += jump
459  * (integrator.GetGlobalWeight(i) * face.sharer_height());
460  }
461  }
462  for (Face const &face : part_ptr()->GetGhostFaces()) {
463  const auto &holder = face.holder();
464  const auto &sharer = face.sharer();
465  const auto &integrator = face.integrator();
466  for (int i = 0; i < integrator.CountPoints(); ++i) {
467  Value jump = spatial_ptr_->GetValueJump(face, i);
468  jump.array() *= jump.array();
469  jump_integrals.at(holder.id()) += jump
470  * (integrator.GetGlobalWeight(i) * face.holder_height());
471  }
472  }
473  assert(jump_integrals.size() == part().CountLocalCells());
474  return jump_integrals;

477  static std::vector<Value> GetViscosityValues(
478  std::vector<Value> const &jump_integrals,
479  std::vector<DampingMatrix> const &damping_matrices) {
480  min_dt_ = 1.e+100;
481  std::vector<Value> viscosity_values;
482  viscosity_values.reserve(part().CountLocalCells());
483  for (Cell *curr_cell : part_ptr()->GetLocalCellPointers()) {
484  auto &viscosity_on_curr_cell = viscosity_values.emplace_back();
485  auto &jump_integral_on_curr_cell = jump_integrals.at(curr_cell->id());
486  auto &damping_matrix_on_curr_cell = damping_matrices.at(curr_cell->id());
487  auto const &coeff = curr_cell->polynomial().coeff();
488  assert(coeff.rows() == Cell::K);
489  assert(coeff.cols() == Cell::N);
490  auto GetMaximumSpeedAndReferenceValueSquare = [](Cell const &cell)
491  -> std::pair<Scalar, Value> {
492  Scalar speed, max_speed{ 1e-5/* avoid 0 in divisor */ };
493  typename Convection::Conservative squares,
494  max_squares{ Value::Ones() * 1e-5/* avoid 0 in divisor */ };
495  auto const &polynomial = cell.polynomial();
496  for (int i = 0; i < Cell::N; ++i) {
497  speed = Convection::GetMaximumSpeedAndReferenceValueSquare(
498  polynomial.GetValue(i), &squares);
499  max_speed = std::max(max_speed, speed);
500  mini::algebra::Maximize(&max_squares, squares);
501  }
502  return { max_speed, max_squares };
503  };
504  auto [max_speed, reference_value_square]
505  = GetMaximumSpeedAndReferenceValueSquare(*curr_cell);
506  Scalar cell_length = curr_cell->length();
507  min_dt_ = std::min(min_dt_, cell_length / max_speed);
508  Scalar max_viscosity = max_speed * cell_length / Cell::P;
509  assert(!std::isinf(max_viscosity) && !std::isnan(max_viscosity));
510  Scalar time_base = cell_length / max_speed;
511  reference_value_square *= std::pow(cell_length,
512  (Cell::P + 1)/* accuracy */ + 3/* volume */);
513  // Now, reference_value_square[k] == |u_k|^2 * h_j^{P + 1} * |E_j|
514  for (int k = 0; k < Cell::K; ++k) {
515  if (jump_integral_on_curr_cell[k] < reference_value_square[k]) {
516  viscosity_on_curr_cell[k] = 0.;
517  continue;
518  }
519  auto const &u_row = coeff.row(k);
520  auto const &u_col = u_row.transpose();
521  Scalar damping_rate = -u_row.dot(damping_matrix_on_curr_cell * u_col);
522  assert(damping_rate >= 0);
523  Scalar damping_time = time_base * GetTimeScale();
524  viscosity_on_curr_cell[k] = std::min(max_viscosity, std::max(0.0,
525  // to protect against damping_rate == -0.0, which would lead to -inf
526  jump_integral_on_curr_cell[k] / (damping_rate * damping_time)));
527  min_dt_ = std::min(min_dt_, cell_length / (max_speed
528  + GetSpatialFactor() * viscosity_on_curr_cell[k] / cell_length));
529 #ifndef NDEBUG
530  if (std::isinf(viscosity_on_curr_cell[k])) {
531  throw std::runtime_error("inf viscosity_on_curr_cell[k]: "
532  + std::to_string(jump_integral_on_curr_cell[k]) + " / "
533  + std::to_string(damping_rate * damping_time));
534  }
535 #endif
536  }
537 #ifndef NDEBUG
538  std::fstream log{ "damping" + std::to_string(curr_cell->metis_id) + ".txt", log.out };
539  log << std::scientific << std::setprecision(2) << damping_matrix_on_curr_cell << "\n";
540 #endif
541  }
542  assert(viscosity_values.size() == part().CountLocalCells());
543  return viscosity_values;
544  }
