70  static std::pair<Scalar, Scalar> GetViscosity(Scalar rho, Scalar nu) {
71  Scalar mu = nu * rho;
72  Scalar zeta = -2.0 / 3 * mu; // Stokes' hypothesis
73  return {mu, zeta};
74  }

76  static Scalar GetThermalConductivity(Scalar rho, Property const &property) {
77  Scalar mu = property.nu * rho;
78  return mu * Gas::Cp() / property.prandtl;
79  }

82  static constexpr int kMass = 0;
83  static constexpr int U = 1 + X;

90  public:
91  static std::pair<Primitive, Gradient> ConservativeToPrimitive(
92  Conservative const &conservative, Gradient const &grad_conservative) {
93  auto primitive = Gas::ConservativeToPrimitive(conservative);
94  Gradient grad_primitive;
95  auto &&grad_u = grad_primitive.col(U);
96  auto &&grad_v = grad_primitive.col(V);
97  auto &&grad_w = grad_primitive.col(W);
98  auto &grad_rho = grad_conservative.col(kMass);
99  auto &grad_rho_u = grad_conservative.col(U);
100  auto &grad_rho_v = grad_conservative.col(V);
101  auto &grad_rho_w = grad_conservative.col(W);
102  grad_primitive.col(kMass) = grad_rho;
103  auto rho = primitive.rho();
104  auto u = primitive.u();
105  auto v = primitive.v();
106  auto w = primitive.w();
107  auto p = primitive.p();
108  grad_u = (grad_rho_u - u * grad_rho) / rho;
109  grad_v = (grad_rho_v - v * grad_rho) / rho;
110  grad_w = (grad_rho_w - w * grad_rho) / rho;
111  auto rho_u = conservative.momentumX();
112  auto rho_v = conservative.momentumY();
113  auto rho_w = conservative.momentumZ();
114  auto &&grad_p = grad_primitive.col(kPressure);
115  grad_p = u * grad_rho_u + rho_u * grad_u;
116  grad_p += v * grad_rho_v + rho_v * grad_v;
117  grad_p += w * grad_rho_w + rho_w * grad_w;
118  grad_p *= -0.5;
119  grad_p += grad_conservative.col(kEnergy);
120  grad_p *= Gas::GammaMinusOne();
121  auto &&grad_T = grad_primitive.col(kTemperature);
122  grad_T = grad_p / rho - (p / (rho * rho)) * grad_rho;
123  grad_T /= Gas::R();
124  return {primitive, grad_primitive};
125  }

127  static Tensor GetViscousStressTensor(Gradient const &grad_primitive,
128  Scalar rho, Scalar nu) {
129  Tensor tau;
130  const auto &grad_u = grad_primitive.col(U);
131  const auto &grad_v = grad_primitive.col(V);
132  const auto &grad_w = grad_primitive.col(W);
133  auto div_uvw = grad_u[X] + grad_v[Y] + grad_w[Z];
134  auto [mu, zeta] = GetViscosity(rho, nu);
135  tau[XX] = 2 * mu * grad_u[X] + zeta * div_uvw;
136  tau[YY] = 2 * mu * grad_v[Y] + zeta * div_uvw;
137  tau[ZZ] = 2 * mu * grad_w[Z] + zeta * div_uvw;
138  tau[XY] = mu * (grad_u[Y] + grad_v[X]);
139  tau[YZ] = mu * (grad_v[Z] + grad_w[Y]);
140  tau[ZX] = mu * (grad_w[X] + grad_u[Z]);
141  return tau;
142  }

149  public:
150  static void MinusViscousFlux(FluxMatrix *flux, Property const &property,
151  Conservative const &conservative, Gradient const &grad_conservative) {
152  auto [primitive, grad_primitive]
153  = ConservativeToPrimitive(conservative, grad_conservative);
154  Tensor tau = GetViscousStressTensor(grad_primitive, conservative.mass(),
155  property.nu);
156  Scalar kappa = GetThermalConductivity(conservative.mass(), property);
157  auto const &uvw = primitive.velocity();
158  auto const &grad_T = grad_primitive.col(kTemperature);
159  flux->row(kEnergy) -= kappa * grad_T;
160  auto &&flux_x = flux->col(X);
161  flux_x[U] -= tau[XX];
162  flux_x[V] -= tau[XY];
163  flux_x[W] -= tau[XZ];
164  flux_x[kEnergy] -= Dot(tau[XX], tau[XY], tau[XZ], uvw);
165  auto &&flux_y = flux->col(Y);
166  flux_y[U] -= tau[YX];
167  flux_y[V] -= tau[YY];
168  flux_y[W] -= tau[YZ];
169  flux_y[kEnergy] -= Dot(tau[YX], tau[YY], tau[YZ], uvw);
170  auto &&flux_z = flux->col(Z);
171  flux_z[U] -= tau[ZX];
172  flux_z[V] -= tau[ZY];
173  flux_z[W] -= tau[ZZ];
174  flux_z[kEnergy] -= Dot(tau[ZX], tau[ZY], tau[ZZ], uvw);
175  }

178  static void _MinusViscousFlux(Flux *flux, Property const &property,
179  Scalar rho, Gradient const &grad_primitive,
180  Vector const &normal, Vector const &uvw, Scalar normal_dot_grad_T) {
181  Tensor tau = GetViscousStressTensor(grad_primitive, rho, property.nu);
182  (*flux)[U] -= Dot(tau[XX], tau[XY], tau[XZ], normal);
183  (*flux)[V] -= Dot(tau[YX], tau[YY], tau[YZ], normal);
184  (*flux)[W] -= Dot(tau[ZX], tau[ZY], tau[ZZ], normal);
185  Scalar kappa = GetThermalConductivity(rho, property);
186  Scalar work_x = Dot(tau[XX], tau[XY], tau[XZ], uvw);
187  Scalar work_y = Dot(tau[YX], tau[YY], tau[YZ], uvw);
188  Scalar work_z = Dot(tau[ZX], tau[ZY], tau[ZZ], uvw);
189  (*flux)[kEnergy] -= Dot(work_x, work_y, work_z, normal)
190  + kappa * normal_dot_grad_T;
191  }

204  static void MinusViscousFluxOnNoSlipWall(Flux *flux,
205  Property const &property, Value const &wall_value,
206  Conservative const &conservative, Gradient const &grad_conservative,
207  Vector const &normal, Scalar value_penalty) {
208  auto [primitive, grad_primitive]
209  = ConservativeToPrimitive(conservative, grad_conservative);
210  auto const &wall_value_ref = static_cast<Primitive const &>(wall_value);
211  auto const &uvw = wall_value_ref.velocity();
212  Vector penalty = value_penalty * (uvw - primitive.velocity());
213  grad_primitive(X, U) += normal[X] * penalty[X];
214  grad_primitive(X, V) += normal[X] * penalty[Y];
215  grad_primitive(X, W) += normal[X] * penalty[Z];
216  grad_primitive(Y, U) += normal[Y] * penalty[X];
217  grad_primitive(Y, V) += normal[Y] * penalty[Y];
218  grad_primitive(Y, W) += normal[Y] * penalty[Z];
219  grad_primitive(Z, U) += normal[Z] * penalty[X];
220  grad_primitive(Z, V) += normal[Z] * penalty[Y];
221  grad_primitive(Z, W) += normal[Z] * penalty[Z];
222  _MinusViscousFlux(flux, property, conservative.mass(), grad_primitive, normal, uvw,
223  wall_value_ref.energy());
224  }
