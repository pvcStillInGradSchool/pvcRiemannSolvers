49  static int GetFaceId(const size_t *cell_nodes, size_t *face_nodes,
50  int face_n_node/* number of nodes on triangle */) {
51  int cnt[5] = { 0, 0, 0, 0, 1 };
52  for (int f = 0; f < face_n_node; ++f) {
53  for (int c = 0; c < 4; ++c) {
54  if (cell_nodes[c] == face_nodes[f]) {
55  cnt[c] = 1;
56  break;
57  }
58  }
59  }
60  assert(3 == std::accumulate(cnt, cnt + 5, 0));
61  int i_face;
62  if (cnt[0]) {
63  if (cnt[1]) {
64  i_face = 0;
65  } else {
66  assert(cnt[3]);
67  i_face = 3;
68  }
69  } else {
70  assert(cnt[2]);
71  if (cnt[1]) {
72  i_face = 1;
73  } else {
74  assert(cnt[3]);
75  i_face = 2;
76  }
77  }
78  return i_face;
79  }

142  static void LocalToShapeGradients(Scalar x_local, Scalar y_local,
143  Scalar z_local, Local *grads) {
144  std::array<Scalar, kNodes> factor_xy, factor_yz, factor_zx;
145  for (int i = 0; i < 4; ++i) {

147  auto factor_x = (1 + local_i[X] * x_local);
148  auto factor_y = (1 + local_i[Y] * y_local);

151  factor_yz[i] = factor_y * factor_z / 8;
152  factor_zx[i] = factor_z * factor_x / 8;
153  }
154  for (int i = 0; i < 4; ++i) {
155  auto &local_i = local_coords_[i];
156  auto &grad_i = grads[i];
157  grad_i[X] = local_i[X] * factor_yz[i];
158  grad_i[Y] = local_i[Y] * factor_zx[i];
159  grad_i[Z] = local_i[Z] * factor_xy[i];
160  }
161  grads[4][X] = grads[4][Y] = 0;
162  grads[4][Z] = 0.5;
163  }
