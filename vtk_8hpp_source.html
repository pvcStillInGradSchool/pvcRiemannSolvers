22 void Encode(Byte *out, Byte const *in) {
23  out[0] = base[((in[0] & 0xFC/* 1111 1100 */) >> 2)];
24  out[1] = base[((in[0] & 0x03/* 0000 0011 */) << 4)
25  + ((in[1] & 0xF0/* 1111 0000 */) >> 4)];
26  out[2] = base[((in[1] & 0x0F/* 0000 1111 */) << 2)
27  + ((in[2] & 0xC0/* 1100 0000 */) >> 6)];
28  out[3] = base[((in[2] & 0x3F/* 0011 1111 */))];
29 }

31 int CharToIndex(char c) {
32  int i = -1;
33  if ('A' <= c && c <= 'Z') {
34  i = c - 'A';
35  } else if ('a' <= c && c <= 'z') {
36  i = c - 'a' + 26;
37  } else if ('0' <= c && c <= '9') {
38  i = c - '0' + 52;
39  } else if ('+' == c) {
40  i = 63;
41  } else {
42  assert('=' == c);
43  }
44  assert(i == -1 || base[i] == c);
45  return i;
46 }

48 void Decode(Byte *out, Byte const *in) {
49  int i_0 = CharToIndex(in[0]);
50  int i_1 = CharToIndex(in[1]);
51  int i_2 = CharToIndex(in[2]);
52  int i_3 = CharToIndex(in[3]);
53  out[0] = ((i_0/* & 0x3F 0011 1111 */) << 2)
54  + ((i_1 & 0x30/* 0011 0000 */) >> 4);
55  out[1] = ((i_1 & 0x0F/* 0000 1111 */) << 4)
56  + ((i_2 & 0x3C/* 0011 1100 */) >> 2);
57  out[2] = ((i_2 & 0x03/* 0000 0011 */) << 6)
58  + ((i_3/* & 0x3F 0011 1111 */));
59 }

61 std::string EncodeBase64(Byte const *input_data, std::size_t n_char) {
62  auto output = std::string();
63  output.resize((n_char / 3 + (n_char % 3 != 0)) * 4);
64  auto *output_data = output.data();
65  auto *output_end = output_data + output.size();
66  auto *input_end = input_data + n_char;
67  for (std::size_t i_triplet = 0, n_triplet = n_char / 3;
68  i_triplet < n_triplet; i_triplet++) {
69  // for each (char[3]) input_data, convert it to (char[4]) output_data:
70  Encode(output_data, input_data);
71  input_data += 3;
72  output_data += 4;
73  }
74  assert(input_data <= input_end);
75  if (input_data != input_end) { // need padding
76  Byte triplet[3] = { 0, 0, 0 };
77  int i = 0;
78  while (input_data != input_end) {
79  triplet[i++] = *input_data++;
80  }
81  assert(i == 1 || i == 2);
82  i++; // 1 in input -> 2 in output, 2 in input -> 3 in output
83  Encode(output_data, triplet);
84  output_data += i;
85  while (output_data != output_end) {
86  *output_data++ = '=';
87  i++;
88  }
89  assert(i == 4);
90  }
91  return output;
92 }

101 void DecodeBase64(std::string const &input, std::size_t n_byte,
102  Byte *output_data) {
103  auto *output_end = output_data + n_byte;
104  auto *input_data = input.data();
105  auto *input_end = input_data + input.size();
106  for (std::size_t i_triplet = 0, n_triplet = n_byte / 3;
107  i_triplet < n_triplet; i_triplet++) {
108  // for each (char[4]) input_data, convert it to (char[3]) output_data:
109  Decode(output_data, input_data);
110  input_data += 4;
111  output_data += 3;
112  }
113  assert(input_data <= input_end);
114  if (output_data != output_end) { // has padding
115  Byte triplet[3];
116  Decode(triplet, input_data);
117  for (int i = 0; output_data != output_end; ++i) {
118  *output_data++ = triplet[i];
119  }
120  }

463  locals = &Hexahedron27<Coord>::locals[0];
464  n = 27;

466  case CellType::kHexahedron64:
