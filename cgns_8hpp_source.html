1021  for (auto &sect : sections_) {

1025  // TODO(PVC): use std::ranges::view in C++23
1026  std::iota(order.begin(), order.end(), 0);
1027  sort(order.begin(), order.end(), [&dim_then_oldid](int lid, int rid) {
1028  return dim_then_oldid[lid].first > dim_then_oldid[rid].first || (
1029  dim_then_oldid[lid].first == dim_then_oldid[rid].first &&
1030  dim_then_oldid[lid].second < dim_then_oldid[rid].second);
1031  });
1032  auto sorted_sections = std::vector<std::unique_ptr<Section<Real>>>(n);
1033  for (int i = 0; i < n; ++i) {
1034  std::swap(sorted_sections[i], sections_[order[i]]);
1035  }
1036  std::swap(sorted_sections, sections_);
1037  int i_next = 1;
1038  for (int i_sect = 1; i_sect <= n; ++i_sect) {
1039  auto &sect = *sections_[i_sect-1];
1040  sect.i_sect_ = i_sect;
