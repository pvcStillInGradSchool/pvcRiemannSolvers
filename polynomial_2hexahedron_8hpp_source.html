172  void InitializeJacobian() requires(kLocal) {
173  for (int ijk = 0; ijk < N; ++ijk) {
174  auto &local = integrator_ptr_->GetLocal(ijk);
175  Jacobian mat = coordinate().LocalToJacobian(local);
176  Jacobian inv = mat.inverse();
177  Scalar det = mat.determinant();
178  jacobian_det_[ijk] = det;
179  jacobian_det_inv_[ijk] = det * inv;
180  jacobian_det_grad_[ijk]
181  = coordinate().LocalToJacobianDeterminantGradient(local);
182  // cache for evaluating Hessian
183  Jacobian inv_T = inv.transpose();
184  mat_after_hess_of_U_[ijk] = inv_T / det;
185  auto mat_grad = coordinate().LocalToJacobianGradient(local);
186  Jacobian inv_T_grad[3];
187  inv_T_grad[X] = -(inv * mat_grad[X] * inv).transpose();
188  inv_T_grad[Y] = -(inv * mat_grad[Y] * inv).transpose();
189  inv_T_grad[Z] = -(inv * mat_grad[Z] * inv).transpose();
190  Mat3x1 const &det_grad = jacobian_det_grad_[ijk];
191  Scalar det2 = det * det;
192  mat_after_grad_of_U_[ijk][X] = inv_T_grad[X] / det
193  + inv_T * (-det_grad[X] / det2);
194  mat_after_grad_of_U_[ijk][Y] = inv_T_grad[Y] / det
195  + inv_T * (-det_grad[Y] / det2);
196  mat_after_grad_of_U_[ijk][Z] = inv_T_grad[Z] / det
197  + inv_T * (-det_grad[Z] / det2);
198  mat_before_grad_of_U_[ijk] = det_grad.transpose() * inv_T / det2;
199  auto det_hess = coordinate().LocalToJacobianDeterminantHessian(local);
200  auto &mat_before_U = mat_before_U_[ijk];
201  mat_before_U(X, X) = det_hess[XX];
202  mat_before_U(X, Y) = det_hess[XY];
203  mat_before_U(X, Z) = det_hess[XZ];
204  mat_before_U(Y, X) = det_hess[YX];
205  mat_before_U(Y, Y) = det_hess[YY];
206  mat_before_U(Y, Z) = det_hess[YZ];
207  mat_before_U(Z, X) = det_hess[ZX];
208  mat_before_U(Z, Y) = det_hess[ZY];
209  mat_before_U(Z, Z) = det_hess[ZZ];
210  mat_before_U *= inv_T / det2;
211  Scalar det3 = det2 * det;
212  mat_before_U.row(X) += det_grad.transpose() *
213  (inv_T_grad[X] / det2 + inv_T * (-2 * det_grad[X] / det3));
214  mat_before_U.row(Y) += det_grad.transpose() *
215  (inv_T_grad[Y] / det2 + inv_T * (-2 * det_grad[Y] / det3));
216  mat_before_U.row(Z) += det_grad.transpose() *
217  (inv_T_grad[Z] / det2 + inv_T * (-2 * det_grad[Z] / det3));
218  }
219  }

221  void InitializeJacobian() requires(!kLocal) {
222  for (int ijk = 0; ijk < N; ++ijk) {
223  auto &local = integrator_ptr_->GetLocal(ijk);
224  Jacobian jacobian = coordinate().LocalToJacobian(local);
225  Jacobian inv = jacobian.inverse();
226  basis_global_gradients_[ijk] = inv * basis_local_gradients_[ijk];
227  // cache for evaluating Hessian
228  Scalar det = jacobian.determinant();
229  jacobian_det_[ijk] = det;
230  jacobian_det_inv_[ijk] = det * inv;
231  jacobian_inv_[ijk] = inv;
232  mat_after_hess_of_U_[ijk] = inv.transpose();
233  auto mat_grad = coordinate().LocalToJacobianGradient(local);
234  Jacobian (&inv_T_grad)[3] = mat_after_grad_of_U_[ijk];
235  inv_T_grad[X] = -(inv * mat_grad[X] * inv).transpose();
236  inv_T_grad[Y] = -(inv * mat_grad[Y] * inv).transpose();
237  inv_T_grad[Z] = -(inv * mat_grad[Z] * inv).transpose();
238  }

496  Hessian _GetGlobalHessian(Gradient const &local_grad_ijk, int ijk) const
497  requires(!kLocal) {
498  Hessian local_hess = GetLocalHessian(ijk);
499  auto &global_hess = local_hess;
500  for (int k = 0; k < K; ++k) {
501  Mat3x3 scalar_hess;
502  scalar_hess(X, X) = local_hess(XX, k);
503  scalar_hess(X, Y) =
504  scalar_hess(Y, X) = local_hess(XY, k);
505  scalar_hess(X, Z) =
506  scalar_hess(Z, X) = local_hess(XZ, k);
507  scalar_hess(Y, Y) = local_hess(YY, k);
508  scalar_hess(Y, Z) =
509  scalar_hess(Z, Y) = local_hess(YZ, k);
510  scalar_hess(Z, Z) = local_hess(ZZ, k);
511  scalar_hess *= mat_after_hess_of_U_[ijk];
512  Mat1x3 scalar_local_grad = local_grad_ijk.col(k);
513  scalar_hess.row(X) += scalar_local_grad * mat_after_grad_of_U_[ijk][X];
514  scalar_hess.row(Y) += scalar_local_grad * mat_after_grad_of_U_[ijk][Y];
515  scalar_hess.row(Z) += scalar_local_grad * mat_after_grad_of_U_[ijk][Z];
516  scalar_hess = jacobian_inv_[ijk] * scalar_hess;
517  global_hess(XX, k) = scalar_hess(X, X);
518  global_hess(XY, k) = scalar_hess(X, Y);
519  global_hess(XZ, k) = scalar_hess(X, Z);
520  global_hess(YY, k) = scalar_hess(Y, Y);
521  global_hess(YZ, k) = scalar_hess(Y, Z);
522  global_hess(ZZ, k) = scalar_hess(Z, Z);
523  }
524  return global_hess;
525  }

527  Hessian _GetGlobalHessian(Gradient const &local_grad_ijk, int ijk) const
528  requires(kLocal) {
529  Hessian local_hess = GetLocalHessian(ijk);
530  auto &global_hess = local_hess;
531  for (int k = 0; k < K; ++k) {
532  Mat3x3 scalar_hess;
533  scalar_hess(X, X) = local_hess(XX, k);
534  scalar_hess(X, Y) =
535  scalar_hess(Y, X) = local_hess(XY, k);
536  scalar_hess(X, Z) =
537  scalar_hess(Z, X) = local_hess(XZ, k);
538  scalar_hess(Y, Y) = local_hess(YY, k);
539  scalar_hess(Y, Z) =
540  scalar_hess(Z, Y) = local_hess(YZ, k);
541  scalar_hess(Z, Z) = local_hess(ZZ, k);
542  scalar_hess *= mat_after_hess_of_U_[ijk];
543  Mat1x3 scalar_local_grad = local_grad_ijk.col(k);
544  scalar_hess.row(X) += scalar_local_grad * mat_after_grad_of_U_[ijk][X]
545  - mat_before_grad_of_U_[ijk] * scalar_local_grad[X];
546  scalar_hess.row(Y) += scalar_local_grad * mat_after_grad_of_U_[ijk][Y]
547  - mat_before_grad_of_U_[ijk] * scalar_local_grad[Y];
548  scalar_hess.row(Z) += scalar_local_grad * mat_after_grad_of_U_[ijk][Z]
549  - mat_before_grad_of_U_[ijk] * scalar_local_grad[Z];
550  Scalar scalar_local_val = this->coeff_(k, ijk);
551  scalar_hess -= mat_before_U_[ijk] * scalar_local_val;
552  scalar_hess = jacobian_det_inv_[ijk] * scalar_hess;
553  scalar_hess /= jacobian_det_[ijk];
554  global_hess(XX, k) = scalar_hess(X, X);
555  global_hess(XY, k) = scalar_hess(X, Y);
556  global_hess(XZ, k) = scalar_hess(X, Z);
557  global_hess(YY, k) = scalar_hess(Y, Y);
558  global_hess(YZ, k) = scalar_hess(Y, Z);
559  global_hess(ZZ, k) = scalar_hess(Z, Z);
560  }
561  return global_hess;
