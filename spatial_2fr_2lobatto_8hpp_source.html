72  template <std::ranges::input_range Range, class FaceToCell>
73  void CacheCorrectionGradients(Range &&faces, FaceToCell &&face_to_cell,
74  std::vector<FaceCache> *cache) {
75  Scalar g_prime = this->vincent_.LocalToRightDerivative(1);
76  for (const Face &face : faces) {
77  assert(cache->size() == face.id());
78  auto &curr_face = cache->emplace_back();
79  const auto &face_integrator = face.integrator();
80  const auto &cell = face_to_cell(face);
81  const auto &cell_integrator = cell.integrator();
82  const auto &cell_polynomial = cell.polynomial();
83  const auto &cell_basis = cell_polynomial.interpolation().basis();
84  int i_face = cell_polynomial.FindFaceId(face.coordinate().center());
85  assert(kFaceQ == face.integrator().CountPoints());
86  for (int f = 0; f < kFaceQ; ++f) {
87  Global const &face_normal = face_integrator.GetNormalFrame(f)[0];
88  auto &flux_point = curr_face.at(f);
89  auto &flux_point_coord = face_integrator.GetGlobal(f);
90  auto [i, j, k] = cell_polynomial.FindCollinearIndex(flux_point_coord, i_face);
91  switch (i_face) {
92  case 0:
93  assert(k == -1);
94  flux_point.ijk = cell_basis.index(i, j, 0);
95  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
96  flux_point.normal =
97  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Z);
98  flux_point.xyz = Z;
99  assert(Collinear(face_normal, flux_point.normal));
100  flux_point.scale = -flux_point.normal.norm();
101  flux_point.g_prime = -g_prime;
102  break;
103  case 1:
104  assert(j == -1);
105  flux_point.ijk = cell_basis.index(i, 0, k);
106  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
107  flux_point.normal =
108  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Y);
109  flux_point.xyz = Y;
110  assert(Collinear(face_normal, flux_point.normal));
111  flux_point.scale = -flux_point.normal.norm();
112  flux_point.g_prime = -g_prime;
113  break;
114  case 2:
115  assert(i == -1);
116  flux_point.ijk = cell_basis.index(kLineQ - 1, j, k);
117  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
118  flux_point.normal =
119  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(X);
120  flux_point.xyz = X;
121  assert(Collinear(face_normal, flux_point.normal));
122  flux_point.scale = +flux_point.normal.norm();
123  flux_point.g_prime = +g_prime;
124  break;
125  case 3:
126  assert(j == -1);
127  flux_point.ijk = cell_basis.index(i, kLineQ - 1, k);
128  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
129  flux_point.normal =
130  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Y);
131  flux_point.xyz = Y;
132  assert(Collinear(face_normal, flux_point.normal));
133  flux_point.scale = +flux_point.normal.norm();
134  flux_point.g_prime = +g_prime;
135  break;
136  case 4:
137  assert(i == -1);
138  flux_point.ijk = cell_basis.index(0, j, k);
139  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
140  flux_point.normal =
141  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(X);
142  flux_point.xyz = X;
143  assert(Collinear(face_normal, flux_point.normal));
144  flux_point.scale = -flux_point.normal.norm();
145  flux_point.g_prime = -g_prime;
146  break;
147  case 5:
148  assert(k == -1);
149  flux_point.ijk = cell_basis.index(i, j, kLineQ - 1);
150  assert(Near(flux_point_coord, cell_integrator.GetGlobal(flux_point.ijk)));
151  flux_point.normal =
152  cell_polynomial.GetJacobianAssociated(flux_point.ijk).col(Z);
153  flux_point.xyz = Z;
154  assert(Collinear(face_normal, flux_point.normal));
155  flux_point.scale = +flux_point.normal.norm();
156  flux_point.g_prime = +g_prime;
157  break;
158  default:
159  assert(false);
160  }
161  }
162  }

165  void CheckCacheConsistency(std::vector<FaceCache> const &this_cache,
166  std::vector<typename Base::FaceCache> const &base_cache) const {
167  int n = this_cache.size();
168  for (int i = 0; i < n; ++i) {
169  auto &face_cache = this_cache.at(i);
170  auto &face_cache_base = base_cache.at(i);
171  for (int f = 0; f < kFaceQ; ++f) {
172  auto &flux_point = face_cache[f];
173  auto &[solution_points, flux_point_base] = face_cache_base[f];
174  assert(flux_point.ijk == flux_point_base.ijk);
175  assert(flux_point.xyz == flux_point_base.xyz);
176  assert(flux_point.scale == flux_point_base.scale);
177  assert(flux_point.normal == flux_point_base.normal);
178  int n_match = 0;
179  assert(kLineQ == solution_points.size());
180  for (int g = 0; g < kLineQ; ++g) {
181  if (flux_point.ijk == solution_points[g].ijk) {
182  n_match++;
183  assert(solution_points[g].g_prime == flux_point.g_prime);
184  } else {
185  assert(std::abs(solution_points[g].g_prime) < 1e-8);
186  }
187  }
188  assert(n_match == 1);
189  }
190  }

237  assert(kFaceQ == face.integrator().CountPoints());

239  auto &holder_flux_point = holder_cache[f];
240  auto &sharer_flux_point = sharer_cache[f];
241  auto [f_holder, f_sharer] = Base::GetFluxForHolderAndSharer(riemanns[f],
242  face.holder(), holder_flux_point,
243  face.sharer(), sharer_flux_point, sharer_data);
244  f_holder *= holder_flux_point.g_prime;
245  assert(holder_data);
246  Polynomial::MinusValue(f_holder, holder_data, holder_flux_point.ijk);
247  if (nullptr == sharer_data) { continue; }
248  f_sharer *= sharer_flux_point.g_prime;
249  Polynomial::MinusValue(f_sharer, sharer_data, sharer_flux_point.ijk);
