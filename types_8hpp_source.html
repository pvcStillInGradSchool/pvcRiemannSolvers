288  template <int kDimensions>
289  static Primitives<Scalar, kDimensions> ConservativeToPrimitive(
290  Conservatives<Scalar, kDimensions> const &conservative) {
291  auto primitive = Primitives<Scalar, kDimensions>(conservative);
292  SetZeroIfNegative(&primitive);
293  if (primitive.rho()) {
294  Tuple<Scalar, kDimensions>::MomentumToVelocity(primitive.rho(),
295  &primitive);
296  primitive.energy() -= primitive.GetDynamicPressure();
297  primitive.energy() *= GammaMinusOne();
298  SetZeroIfNegative(&primitive);
299  }
300  return primitive;
301  }
302  template <int kDimensions>
303  static Conservatives<Scalar, kDimensions> PrimitiveToConservative(
304  Primitives<Scalar, kDimensions> const &primitive) {
305  auto conservative = Conservatives<Scalar, kDimensions>(primitive);
306  Tuple<Scalar, kDimensions>::VelocityToMomentum(primitive.rho(),
307  &conservative);
308  conservative.energy() *= OneOverGammaMinusOne(); // p / (gamma - 1)
309  conservative.energy() += primitive.GetDynamicPressure();
310  return conservative;
311  }
312  template <int kDimensions>
313  static FluxTuple<Scalar, kDimensions> PrimitiveToFlux(
314  const Primitives<Scalar, kDimensions> &primitive) {
315  auto conservative = PrimitiveToConservative(primitive);
316  conservative *= primitive.u();
317  auto flux_x = FluxTuple<Scalar, kDimensions>(conservative);
318  flux_x.momentumX() += primitive.p();
319  flux_x.energy() += primitive.p() * primitive.u();
320  return flux_x;
321  }
322  static auto GetFluxMatrix(Conservatives<Scalar, 3> const& cv) {
323  using FluxMatrix = typename FluxTuple<Scalar, 3>::FluxMatrix;
324  FluxMatrix mat;
325  auto pv = ConservativeToPrimitive(cv);
326  auto rho = pv.rho(), u = pv.u(), v = pv.v(), w = pv.w(), p = pv.p();
327  auto rho_u = cv.momentumX(), rho_v = cv.momentumY(), rho_w = cv.momentumZ();
328  auto rho_h0 = cv.energy() + p;
329  mat.col(0) << rho_u, rho_u * u + p, rho_v * u, rho_w * u, rho_h0 * u;
330  mat.col(1) << rho_v, rho_u * v, rho_v * v + p, rho_w * v, rho_h0 * v;
331  mat.col(2) << rho_w, rho_u * w, rho_v * w, rho_w * w + p, rho_h0 * w;
332  return mat;
333  }
