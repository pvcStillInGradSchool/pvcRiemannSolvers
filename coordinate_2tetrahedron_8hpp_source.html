157  using Base = Tetrahedron<Scalar>;
158  static constexpr int AB{4}, BC{5}, CA{6}, DA{7}, DB{8}, DC{9};

168 
169  private:

178  void SortNodesOnFace(const size_t *cell_nodes, size_t *face_nodes,
179  int face_n_node) const final {
180  assert(6 == face_n_node);
181  int i_face;
182  int cnt[4] = { 0, 0, 0, 0 };
183  for (int f = 0; f < 6; ++f) {
184  for (int c = 0; c < 4; ++c) {
185  if (cell_nodes[c] == face_nodes[f]) {
186  cnt[c] = true;
187  break;
188  }
189  }
190  }
191  if (cnt[A]) {
192  if (cnt[B]) {
193  if (cnt[C]) {
194  i_face = 0;
195  } else {
196  assert(cnt[D]);
197  i_face = 1;
198  }
199  } else {
200  assert(cnt[C] && cnt[D]);
201  i_face = 2;
202  }
203  } else {
204  assert(cnt[B] && cnt[C] && cnt[D]);
205  i_face = 3;
206  }
207  for (int i = 0; i < 6; ++i) {
208  face_nodes[i] = cell_nodes[faces_[i_face][i]];
209  }
210  }
211 
212  public:
213  std::vector<Scalar> LocalToShapeFunctions(
214  Scalar a, Scalar b, Scalar c) const final {
215  auto shapes = std::vector<Scalar>(kNodes);
216  auto d = 1.0 - a - b - c;
217  shapes[A] = a * (a - 0.5) * 2;
218  shapes[B] = b * (b - 0.5) * 2;
219  shapes[C] = c * (c - 0.5) * 2;
220  shapes[D] = d * (d - 0.5) * 2;
221  a *= 4;
222  shapes[AB] = b * a/* 4a */;
223  b *= 4;
224  shapes[BC] = c * b/* 4b */;
225  shapes[CA] = c * a/* 4a */;
226  shapes[DA] = d * a/* 4a */;
227  shapes[DB] = d * b/* 4b */;
228  shapes[DC] = d * c * 4;
229  return shapes;
230  }
231  std::vector<Local> LocalToShapeGradients(
232  Scalar a, Scalar b, Scalar c) const final {
233  auto grads = std::vector<Local>(kNodes);
234  auto factor_a = 4 * a;
235  auto factor_b = 4 * b;
236  auto factor_c = 4 * c;
237  auto factor_d = 4 - factor_a - factor_b - factor_c;
238  // shapes[A] = a * (a - 0.5) * 2;
239  grads[A][A] = factor_a - 1;
240  grads[A][B] = 0;
241  grads[A][C] = 0;
242  // shapes[B] = b * (b - 0.5) * 2;
243  grads[B][A] = 0;
244  grads[B][B] = factor_b - 1;
245  grads[B][C] = 0;
246  // shapes[C] = c * (c - 0.5) * 2;
247  grads[C][A] = 0;
248  grads[C][B] = 0;
249  grads[C][C] = factor_c - 1;
250  // shapes[D] = d * (d - 0.5) * 2 = (1 - a - b - c) * (1/2 - a - b - c) * 2;
251  grads[D][A] = grads[D][B] = grads[D][C] = 1 - factor_d;
252  // shapes[AB] = a * b * 4;
253  grads[AB][A] = factor_b;
254  grads[AB][B] = factor_a;
255  grads[AB][C] = 0;
256  // shapes[BC] = b * c * 4;
257  grads[BC][A] = 0;
258  grads[BC][B] = factor_c;
259  grads[BC][C] = factor_b;
260  // shapes[CA] = c * a * 4;
261  grads[CA][A] = factor_c;
262  grads[CA][B] = 0;
263  grads[CA][C] = factor_a;
264  // shapes[DA] = d * a * 4 = (1 - a - b - c) * a * 4;
265  grads[DA][A] = factor_d - factor_a;
266  grads[DA][B] = grads[DA][C] = -factor_a;
267  // shapes[DB] = d * b * 4 = (1 - a - b - c) * b * 4;
268  grads[DB][B] = factor_d - factor_b;
269  grads[DB][A] = grads[DB][C] = -factor_b;
270  // shapes[DC] = d * c * 4 = (1 - a - b - c) * c * 4;
271  grads[DC][C] = factor_d - factor_c;
272  grads[DC][A] = grads[DC][B] = -factor_c;
273  return grads;
274  }
