57  static std::vector<Int> FindForegroundFringeCells(int n_layer,
58  Mesh const &mesh, Graph const &graph, Mapper const &mapper) {
59  std::vector<Int> result;
60  assert(mesh.CountBases() == 1);
61  auto &base = mesh.GetBase(1);
62  // Find the outmost layer according to #neighbors < #faces:
63  for (Int i = 0, n = graph.CountVertices(); i < n; ++i) {
64  auto cell = mapper.metis_to_cgns_for_cells[i];
65  auto &sect = base.GetZone(cell.i_zone).GetSection(cell.i_sect);
66  if (graph.CountNeighbors(i) < sect.CountFacesByType()) {
67  result.push_back(i);
68  }
69  }
70  // Run breadth-first search for more layers:
71  auto curr_begin = result.begin(), curr_end = result.end();
72  auto found = std::unordered_set<Int>(curr_begin, curr_end);
73  while (curr_begin != curr_end && --n_layer > 0) {
74  auto next = std::vector<Int>();
75  while (curr_begin != curr_end) {
76  auto i = *curr_begin++; // for each i in current layer
77  for (auto j : graph.neighbors(i)) {
78  auto iter = found.find(j);
79  if (iter == found.end()) {
80  found.emplace_hint(iter, j);
81  next.push_back(j);
82  }
83  }
84  }
85  result.resize(result.size() + next.size());
86  curr_end = result.end();
87  curr_begin = curr_end - next.size();
88  std::ranges::copy(next, curr_begin);
89  }
90  return result;
91  }

159  static std::vector<std::vector<int>> FindBackgroundDonorCells(
160  Mesh const &mesh_fg, Graph const &graph_fg, Mapper const &mapper_fg,
161  std::vector<Int> const &fringe_fg, Tree const &tree_bg,

163  Real radius) {

165  result.reserve(fringe_fg.size());
166  auto &base = mesh_fg.GetBase(1);
167  for (auto i_cell : fringe_fg) {
168  auto &index = mapper_fg.metis_to_cgns_for_cells[i_cell];
169  auto &sect = base.GetZone(index.i_zone).GetSection(index.i_sect);
170  Real x, y, z;
171  sect.GetCellCenter(index.i_cell, &x, &y, &z);
172  result.emplace_back(tree_bg.Search(x, y, z, 1));
173  // Run breadth-first search for more donor cells:
174  auto &curr = result.back();
175  auto curr_begin = curr.begin(), curr_end = curr.end();
176  auto found = std::unordered_set<Int>(curr_begin, curr_end);
177  while (curr_begin != curr_end) {
178  auto next = std::vector<Int>();
179  while (curr_begin != curr_end) {
180  auto i = *curr_begin++; // for each i in current layer
181  for (auto j : graph_bg.neighbors(i)) {
182  auto iter = found.find(j);
183  if (iter != found.end()) {
184  continue; // skip cells that are already found
185  }
186  auto &index_j = mapper_bg.metis_to_cgns_for_cells[j];
187  auto &zone_j = mesh_bg.GetBase(1).GetZone(index_j.i_zone);

192  continue; // skip cells that are out of the bounding ball
193  }
194  found.emplace_hint(iter, j);

196  }
