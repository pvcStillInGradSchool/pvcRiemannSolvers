28 template <typename Projection, typename Callable>
29 void Project(Projection *proj, Callable &&func) {
30  using Global = typename Projection::Global;
31  using Mat1xN = typename Projection::Mat1xN;
32  using Value = typename Projection::Value;
33  using Coeff = typename Projection::Coeff;
34  using Return = std::invoke_result_t<Callable, Global>;
35  static_assert(std::is_same_v<Return, Value> || std::is_scalar_v<Return>);
36  Coeff coeff = integrator::Integrate([&](Global const &xyz) {
37  auto f_col = func(xyz);
38  Mat1xN b_row = proj->GlobalToBasisValues(xyz);
39  Coeff prod = f_col * b_row;
40  return prod;
41  }, proj->integrator());
42  proj->SetCoeff(coeff);
43 }
