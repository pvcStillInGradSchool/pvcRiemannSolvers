376  def u_exact(self, x_global, t_curr):
377  """ Solve u_curr from u_curr = u_init(x - a(u_curr) * t_curr).
378 
379  Currently, the solution is only correct for standing waves in t <= 10.
380 
381  To product for general cases, it's better to use a higher order scheme with a fine grid, e.g.:
382 
383  python3 solver.py --t_end 2.0 --method FRonUniformRoots --degree 2 --problem Burgers --detector KXRCF --limiter LWA --n_element 201 --output vtu --n_frame 100
384  """
385  if self._u_mean_u_mean:
386  return np.nan
387  def func(u_curr):
388  ku = self._spatial_spatial.equation().get_convective_speed(u_curr)
389  return u_curr - self.u_initu_initu_init(x_global - ku * t_curr)
390  if x_global in self._u_prev_u_prev:
391  u_prev = self._u_prev_u_prev[x_global]
392  else:
393  u_prev = self.u_initu_initu_init(x_global)
394  delta = 0.02
395  u_min = u_prev - delta
396  u_max = u_prev + delta
397  while func(u_min) * func(u_max) > 0:
398  u_min -= delta
399  u_max += delta
400  root = optimize.bisect(func, u_min, u_max)
401  self._u_prev_u_prev[x_global] = root
402  return root
