35 template <typename T>
36 std::pair<std::vector<int>, std::vector<int>>
37 GetNewOrder(T const *parts, int n) {
38  auto new_to_old = std::vector<int>(n);
39  auto old_to_new = std::vector<int>(n);
40  // TODO(PVC): use std::ranges::view in C++23
41  std::iota(new_to_old.begin(), new_to_old.end(), 0);
42  auto cmp = [parts](int lid, int rid) {
43  return parts[lid] < parts[rid] || (parts[lid] == parts[rid] && lid < rid);
44  };
45  std::sort(new_to_old.begin(), new_to_old.end(), cmp);
46  for (int i = 0; i < n; ++i)
47  old_to_new[new_to_old[i]] = i;
48  return {new_to_old, old_to_new};
49 }

86 void ShuffleConnectivity(std::vector<int> const &old_to_new_for_nodes,
87  std::vector<int> const &new_to_old_for_cells,
88  int npe, T *old_cid_old_nid) {
89  int n_cells = new_to_old_for_cells.size();
90  int i_node_list_size = n_cells * npe;
91  auto old_cid_new_nid = std::vector<T>(i_node_list_size);
92  for (int i = 0; i < i_node_list_size; ++i) {
93  auto old_nid = old_cid_old_nid[i];

95  old_cid_new_nid[i] = new_nid;
96  }
97  auto *new_cid_new_nid = old_cid_old_nid;
98  for (int new_cid = 0; new_cid < n_cells; ++new_cid) {
99  int range_min = npe * new_to_old_for_cells[new_cid];
100  auto old_ptr = old_cid_new_nid.data() + range_min;
101  for (int i = 0; i < npe; ++i) {
102  *new_cid_new_nid++ = *old_ptr++;
103  }
104  }

151 template <std::integral Int, std::floating_point Real>
152 void Shuffler<Int, Real>::FillFaceParts(
153  CgnsMesh const &mesh, Mapper const &mapper) {
154  auto &m_to_c_nodes = mapper.metis_to_cgns_for_nodes;
155  auto &m_to_c_cells = mapper.metis_to_cgns_for_cells;
156  auto &c_to_m_nodes = mapper.cgns_to_metis_for_nodes;
157  auto &c_to_m_cells = mapper.cgns_to_metis_for_cells;
158  auto &base = mesh.GetBase(1);
159  auto n_zones = base.CountZones();
160  // For each node, find its user cells and append them to its vector:
161  auto node_user_cells = std::vector<std::vector<Int>>(node_parts_.size());
162  for (int i_zone = 1; i_zone <= n_zones; ++i_zone) {
163  auto &zone = base.GetZone(i_zone);
164  auto &i_node_to_m_node = c_to_m_nodes.at(i_zone);
165  auto n_sects = zone.CountSections();
166  for (int i_sect = 1; i_sect <= n_sects; ++i_sect) {
167  auto &sect = zone.GetSection(i_sect);
168  if (sect.dim() == base.GetCellDim()) {
169  auto npe = sect.CountNodesByType();
170  auto n_cells = sect.CountCells();
171  auto i_cell_min = sect.CellIdMin();
172  auto i_cell_max = sect.CellIdMax();
173  auto &i_cell_to_m_cell = c_to_m_cells.at(i_zone).at(i_sect);
174  auto *head = sect.GetNodeIdList();
175  auto *curr = head;
176  auto *tail = head + n_cells * npe;
177  for (int i_cell = i_cell_min; i_cell <= i_cell_max; ++i_cell) {
178  auto m_cell = i_cell_to_m_cell.at(i_cell);
179  for (int k = 0; k < npe; ++k) {
180  assert(curr + k < tail);
181  auto m_node = i_node_to_m_node.at(curr[k]);
182  node_user_cells[m_node].emplace_back(m_cell);
183  }
184  curr += npe;
185  }
186  assert(curr == tail);
187  }
188  }
189  }
190  // For each face, determine its part by its user cells:
191  for (int i_zone = 1; i_zone <= n_zones; ++i_zone) {
192  auto &zone = base.GetZone(i_zone);
193  auto &i_node_to_m_node = c_to_m_nodes.at(i_zone);
194  auto n_sects = zone.CountSections();
195  for (int i_sect = 1; i_sect <= n_sects; ++i_sect) {
196  auto &sect = zone.GetSection(i_sect);
197  if (sect.dim() + 1 == base.GetCellDim()) {
198  auto npe = sect.CountNodesByType();
199  auto n_faces = sect.CountCells();
200  auto i_face_min = sect.CellIdMin();
201  auto i_face_max = sect.CellIdMax();
202  auto &i_face_to_part = (face_parts_[i_zone][i_sect]
203  = cgns::ShiftedVector<Int>(n_faces, i_face_min));
204  auto *head = sect.GetNodeIdList();
205  auto *curr = head;
206  auto *tail = head + n_faces * npe;
207  // For each face, count the user cells:
208  for (int i_face = i_face_min; i_face <= i_face_max; ++i_face) {
209  auto face_user_cell_cnts = std::unordered_map<Int, Int>();
210  for (int k = 0; k < npe; ++k) {
211  assert(curr + k < tail);
212  auto i_node = curr[k];
213  auto m_node = i_node_to_m_node.at(i_node);
214  for (Int m_cell : node_user_cells[m_node]) {
215  face_user_cell_cnts[m_cell]++;
216  }
217  }
218  std::vector<int> npe_cells;
219  // Find the cell that uses this face npe times:
220  for (auto [m_cell, cnt] : face_user_cell_cnts) {
221  assert(cnt <= npe);
222  if (cnt == npe) {
223  i_face_to_part.at(i_face) = cell_parts_.at(m_cell);
224  npe_cells.emplace_back(m_cell);
225  }
226  }
227  // There should be one and only one such user:
228  assert(npe_cells.size() == 1);
229  if (npe_cells.size() > 1) {
230  for (auto m_cell : npe_cells)
231  std::printf("face[%d] is owned by cell[%d] for %d times.\n",
232  i_face, static_cast<int>(m_cell), npe);
233  }
234  curr += npe;
235  }
236  assert(curr == tail);
237  }
238  }
239  }
240 }

242 template <std::integral Int, std::floating_point Real>
243 void Shuffler<Int, Real>::Shuffle() {
244  FillFaceParts(*cgns_mesh_, *mapper_);
245  auto &m_to_c_nodes = mapper_->metis_to_cgns_for_nodes;
246  auto &m_to_c_cells = mapper_->metis_to_cgns_for_cells;
247  auto &c_to_m_nodes = mapper_->cgns_to_metis_for_nodes;
248  auto &c_to_m_cells = mapper_->cgns_to_metis_for_cells;
249  auto &base = cgns_mesh_->GetBase(1);
250  auto n_zones = base.CountZones();
251  // shuffle nodes, cells and data on them
252  for (int i_zone = 1; i_zone <= n_zones; ++i_zone) {
253  auto &zone = base.GetZone(i_zone);
254  // shuffle node coordinates
255  auto metis_nid_offset = c_to_m_nodes[i_zone][1];
256  auto [new_to_old_for_nodes, old_to_new_for_nodes] = GetNewOrder(
257  &(node_parts_[metis_nid_offset]), zone.CountNodes());
258  auto &coord = zone.GetCoordinates();
259  ShuffleData(new_to_old_for_nodes, coord.x().data());
260  ShuffleData(new_to_old_for_nodes, coord.y().data());
261  ShuffleData(new_to_old_for_nodes, coord.z().data());
262  ShuffleData(new_to_old_for_nodes, &(c_to_m_nodes[i_zone][1]));
263  ShuffleData(old_to_new_for_nodes, &(m_to_c_nodes[metis_nid_offset]));
264  // shuffle data on nodes
265  auto n_solns = zone.CountSolutions();
266  for (auto i_soln = 1; i_soln <= n_solns; i_soln++) {
267  auto &solution = zone.GetSolution(i_soln);
268  if (!solution.OnNodes())
269  continue;
270  for (auto i = 1; i <= solution.CountFields(); ++i) {
271  auto &field = solution.GetField(i);
272  ShuffleData(new_to_old_for_nodes, field.data());
273  }
274  }
275  // shuffle cells and data on cells
276  auto n_sects = zone.CountSections();
277  for (auto i_sect = 1; i_sect <= n_sects; ++i_sect) {
278  auto &section = zone.GetSection(i_sect);
279  /* Shuffle Connectivity */
280  auto npe = section.CountNodesByType();
281  auto n_cells = section.CountCells();
282  auto *i_node_list = section.GetNodeIdList();
283  if (section.dim() != base.GetCellDim()) {
284  // For a lower-dim section, get each face's partition
285  auto [new_to_old_for_faces, old_to_new_for_faces] = GetNewOrder(
286  face_parts_.at(i_zone).at(i_sect).data(), section.CountCells());
287  // then shuffle its connectivity
288  ShuffleConnectivity(old_to_new_for_nodes, new_to_old_for_faces,
289  npe, i_node_list);
290  continue;
291  }
292  auto range_min = section.CellIdMin();
293  auto metis_cid_offset = c_to_m_cells[i_zone][i_sect].at(range_min);
294  auto [new_to_old_for_cells, old_to_new_for_cells] = GetNewOrder(
295  &(cell_parts_[metis_cid_offset]), n_cells);
296  ShuffleData(old_to_new_for_cells, &(m_to_c_cells[metis_cid_offset]));
297  ShuffleData(new_to_old_for_cells, c_to_m_cells[i_zone][i_sect].data());
298  ShuffleConnectivity(old_to_new_for_nodes, new_to_old_for_cells,
299  npe, i_node_list);
300  /* Shuffle Data on Cells */
301  auto n_solns = zone.CountSolutions();
302  for (auto i_soln = 1; i_soln <= n_solns; i_soln++) {
303  auto &solution = zone.GetSolution(i_soln);
304  if (!solution.OnCells())
305  continue;
306  for (auto i = 1; i <= solution.CountFields(); ++i) {
307  auto &field = solution.GetField(i);
308  auto *field_ptr = &(field.at(range_min));
309  ShuffleData(new_to_old_for_cells, field_ptr);
310  }
311  }
312  }
313  zone.UpdateSectionRanges();
314  }
315 }

319  auto &base = cgns_mesh_->GetBase(1);
320  int n_zones = base.CountZones();
321  /* Prepare to-be-written info for each part: */
322  // auto [i_node_min, i_node_max] = part_to_nodes[i_part][i_zone];
323  // auto [i_cell_min, i_cell_max] = part_to_cells[i_part][i_zone][i_sect];

329  = std::vector<std::vector<std::vector<std::pair<int, int>>>>(n_parts_);
330  for (int p = 0; p < n_parts_; ++p) {
331  part_to_nodes[p].resize(n_zones + 1);
332  part_to_cells[p].resize(n_zones + 1);
333  part_to_faces[p].resize(n_zones + 1);
334  for (int z = 1; z <= n_zones; ++z) {
335  part_to_cells[p][z].resize(base.GetZone(z).CountSections() + 1);

353  if (curr_part != prev_part) {

355  prev_nid = curr_nid;
356  prev_part = curr_part;
357  }
358  } // for each node
359  part_to_nodes[prev_part][i_zone] = { prev_nid, n_nodes + 1 };
360  // slice cell lists by i_part
361  int n_cells = zone.CountCells();
362  auto &cell_field = zone.GetSolution("DataOnCells").GetField("PartIndex");
363  for (int i_sect = 1; i_sect <= zone.CountSections(); ++i_sect) {
364  auto &sect = zone.GetSection(i_sect);
365  if (sect.dim() != base.GetCellDim())
366  continue;
367  int cid_min = sect.CellIdMin(), cid_max = sect.CellIdMax();

370  int curr_part = cell_field.at(curr_cid);
371  if (curr_part != prev_part) { // find a new part
372  part_to_cells[prev_part][i_zone][i_sect] = { prev_cid, curr_cid };
373  prev_cid = curr_cid;
374  prev_part = curr_part;
375  }
376  } // for each sell
377  part_to_cells[prev_part][i_zone][i_sect] = { prev_cid, cid_max + 1 };
378  } // for each sect of cells

391  prev_fid = curr_fid;
392  prev_part = curr_part;
393  }
394  } // for each face

396  } // for each sect of faces
397  } // for each zone
