62 template <std::integral Int, std::floating_point Real>
63 typename CgnsToMetis<Int, Real>::MetisMesh
64 CgnsToMetis<Int, Real>::Map(CgnsMesh const &cgns_mesh) {
65  assert(cgns_mesh.CountBases() == 1);
66  auto &base = cgns_mesh.GetBase(1);
67  auto cell_dim = base.GetCellDim();
68  auto n_zones = base.CountZones();
69  Int n_nodes_in_curr_base{0};
70  auto cell_ptr = std::vector<Int>(); // METIS node range
71  auto cell_idx = std::vector<Int>(); // METIS node index
72  Int pointer_value{0};
73  cell_ptr.emplace_back(pointer_value);
74  Int n_nodes_in_prev_zones{0};
75  cgns_to_metis_for_nodes.resize(n_zones+1);
76  cgns_to_metis_for_cells.resize(n_zones+1);
77  for (Int i_zone = 1; i_zone <= n_zones; ++i_zone) {
78  auto &zone = base.GetZone(i_zone);
79  // read nodes in current zone
80  auto n_nodes_in_curr_zone = zone.CountNodes();
81  auto &nodes = cgns_to_metis_for_nodes.at(i_zone);
82  nodes.reserve(n_nodes_in_curr_zone+1);
83  nodes.emplace_back(-1); // nodes[0] is invalid
84  metis_to_cgns_for_nodes.reserve(metis_to_cgns_for_nodes.size() +
85  n_nodes_in_curr_zone);
86  for (Int i_node = 1; i_node <= n_nodes_in_curr_zone; ++i_node) {
87  metis_to_cgns_for_nodes.emplace_back(i_zone, i_node);
88  nodes.emplace_back(n_nodes_in_curr_base++);
89  }
90  // read cells in current zone
91  auto n_sections = zone.CountSections();
92  cgns_to_metis_for_cells[i_zone].resize(n_sections + 1);
93  for (Int i_sect = 1; i_sect <= n_sections; ++i_sect) {
94  auto &section = zone.GetSection(i_sect);
95  if (!zone.CheckTypeDim(section.type(), cell_dim))
96  continue;
97  auto &metis_i_cells = cgns_to_metis_for_cells[i_zone][i_sect];
98  metis_i_cells = ShiftedVector(section.CountCells(), section.CellIdMin());
99  auto n_cells_in_curr_sect = section.CountCells();
100  auto n_nodes_per_cell = section.CountNodesByType();
101  metis_to_cgns_for_cells.reserve(metis_to_cgns_for_cells.size() +
102  n_cells_in_curr_sect);
103  cell_ptr.reserve(cell_ptr.size() + n_cells_in_curr_sect);
104  for (Int i_cell = section.CellIdMin();
105  i_cell <= section.CellIdMax(); ++i_cell) {
106  metis_i_cells.at(i_cell) = metis_to_cgns_for_cells.size();
107  metis_to_cgns_for_cells.emplace_back(i_zone, i_sect, i_cell);
108  cell_ptr.emplace_back(pointer_value += n_nodes_per_cell);
109  }
110  auto i_node_list_size = n_nodes_per_cell * n_cells_in_curr_sect;
111  cell_idx.reserve(cell_idx.size() + i_node_list_size);
112  auto i_node_list = section.GetNodeIdList();
113  for (Int i_node = 0; i_node < i_node_list_size; ++i_node) {
114  auto metis_i_node = n_nodes_in_prev_zones + i_node_list[i_node] - 1;
115  cell_idx.emplace_back(metis_i_node);
116  }
117  }
118  n_nodes_in_prev_zones += zone.CountNodes();
119  }
120  assert(metis_to_cgns_for_nodes.size() == n_nodes_in_curr_base);
121  return MetisMesh(std::move(cell_ptr), std::move(cell_idx),
122  n_nodes_in_curr_base);
123 }

125 template <std::integral Int, std::floating_point Real>
126 bool CgnsToMetis<Int, Real>::IsValid() const {
127  Int metis_n_nodes = metis_to_cgns_for_nodes.size();
128  for (Int metis_i_node = 0; metis_i_node < metis_n_nodes; ++metis_i_node) {
129  auto index = metis_to_cgns_for_nodes.at(metis_i_node);
130  auto &nodes = cgns_to_metis_for_nodes.at(index.i_zone);
131  if (nodes.at(index.i_node) != metis_i_node) {
132  return false;
133  }
134  }
135  Int metis_n_cells = metis_to_cgns_for_cells.size();
136  for (Int metis_i_cell = 0; metis_i_cell < metis_n_cells; ++metis_i_cell) {
137  auto index = metis_to_cgns_for_cells.at(metis_i_cell);
138  auto &cells = cgns_to_metis_for_cells.at(index.i_zone).at(index.i_sect);
139  if (cells.at(index.i_cell) != metis_i_cell) {
140  return false;
141  }
142  }
143  return true;
144 }

146 template <std::integral Int, std::floating_point Real>
147 void CgnsToMetis<Int, Real>::WriteParts(
148  std::vector<idx_t> const &cell_parts,
149  std::vector<idx_t> const &node_parts, CgnsMesh *cgns_mesh) const {
150  auto &zone_to_sects = cgns_to_metis_for_cells;
151  auto &zone_to_nodes = cgns_to_metis_for_nodes;
152  auto &base = cgns_mesh->GetBase(1);
153  int n_zones = base.CountZones();
154  for (int i_zone = 1; i_zone <= n_zones; ++i_zone) {
155  auto &zone = base.GetZone(i_zone);
156  // write data on nodes
157  auto &node_sol = zone.AddSolution("DataOnNodes", CGNS_ENUMV(Vertex));
158  auto &node_field = node_sol.AddField("PartIndex");
159  auto &metis_nids = node_sol.AddField("MetisIndex");
160  auto n_nodes = zone.CountNodes();
161  for (int cgns_nid = 1; cgns_nid <= n_nodes; ++cgns_nid) {
162  auto metis_nid = zone_to_nodes[i_zone][cgns_nid];
163  node_field.at(cgns_nid) = node_parts[metis_nid];
164  metis_nids.at(cgns_nid) = metis_nid;
165  }
166  // write data on cells
167  auto &cell_sol = zone.AddSolution("DataOnCells", CGNS_ENUMV(CellCenter));
168  auto &cell_field = cell_sol.AddField("PartIndex");
169  auto &metis_cids = cell_sol.AddField("MetisIndex");
170  auto &sect_to_cells = zone_to_sects.at(i_zone);
171  auto n_sects = zone.CountSections();
172  for (int i_sect = 1; i_sect <= n_sects; ++i_sect) {
173  auto &sect = zone.GetSection(i_sect);
174  if (sect.dim() != base.GetCellDim())
175  continue;
176  auto &cells_local_to_global = sect_to_cells.at(i_sect);
177  auto n_cells = sect.CountCells();
178  auto range_min{sect.CellIdMin()};
179  auto range_max{sect.CellIdMax()};
180  assert(n_cells - 1 == range_max - range_min);
181  for (int cgns_i_cell = range_min; cgns_i_cell <= range_max;
182  ++cgns_i_cell) {
183  auto metis_i_cell = cells_local_to_global.at(cgns_i_cell);
184  cell_field.at(cgns_i_cell) = cell_parts[metis_i_cell];
185  metis_cids.at(cgns_i_cell) = metis_i_cell;
186  }
187  }
188  }
