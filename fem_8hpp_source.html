97  assert(cell_data_size_ == Cell::kFields * part_ptr->CountLocalCells());
98  auto build_riemanns = [this](std::ranges::input_range auto faces) {
99  for (Face const &face : faces) {
100  assert(face.id() == this->riemann_.size());
101  auto const &integrator = face.integrator();
102  auto &riemanns = this->riemann_.emplace_back(integrator.CountPoints());
103  for (int i = 0, n = riemanns.size(); i < n; ++i) {
104  riemanns.at(i).Rotate(integrator.GetNormalFrame(i));

417  virtual Scalar GetTimeStep(Scalar dt_guess, int rk_order) const {
418  auto GetMaximumSpeed = [](Cell const &cell) -> Scalar {
419  Scalar speed, max_speed{ 1e-5/* avoid 0 in divisor */ };
420  auto const &polynomial = cell.polynomial();
421  for (int i = 0; i < Cell::N; ++i) {
422  speed = Riemann::Convection::GetMaximumSpeed(
423  polynomial.GetValue(i));
424  max_speed = std::max(max_speed, speed);
425  }
426  return max_speed;
427  };
428  Scalar min_dt = 1.e+100;
429  for (Cell const &curr_cell : part().GetLocalCells()) {
430  min_dt = std::min(min_dt,
431  curr_cell.length() / GetMaximumSpeed(curr_cell));
432  }
433  Scalar dt = GetCflNumber(rk_order) * min_dt;
434  return std::min(dt, dt_guess);
