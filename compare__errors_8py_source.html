16 
17  2. Filter out the errors:
18  for x in *.log ; do cat $x | grep error > ${x%.log}.csv ; done
19 
20  3. Replace string:
21  for x in *0.csv ; do sed -i 's/\ ],//g' $x ; done
22  for x in *0.csv ; do sed -i 's/t_curr,\ error_1,\ error_2,\ error_âˆž\ =\ \[\ //g' $x ; done
23  """
24  errors = dict()
