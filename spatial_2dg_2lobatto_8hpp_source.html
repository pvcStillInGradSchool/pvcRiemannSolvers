58  template <std::ranges::input_range Range, class FaceToCell>
59  void MatchIntegratorianPoints(Range &&faces, FaceToCell &&face_to_cell,
60  std::vector<FaceCache> *cache) {
61  for (const Face &face : faces) {
62  assert(cache->size() == face.id());
63  auto &curr_face = cache->emplace_back();
64  const auto &face_integrator = face.integrator();
65  const auto &cell_integrator = face_to_cell(face).integrator();
66  for (int f = 0, F = face_integrator.CountPoints(); f < F; ++f) {
67  auto &flux_point = face_integrator.GetGlobal(f);
68  curr_face.at(f) = -1;
69  for (int h = 0, H = cell_integrator.CountPoints(); h < H; ++h) {
70  if (Near(flux_point, cell_integrator.GetGlobal(h))) {
71  curr_face[f] = h;
72  break;
73  }
74  }
75  assert(curr_face[f] >= 0);
76  }
77  }
78  }

143  Column GetResidualColumn() const override {
144  Column residual = this->Base::GetResidualColumn();
145  // divide mass matrix for each cell
146  for (const Cell &cell : this->part().GetLocalCells()) {
147  auto i_cell = cell.id();
148  Scalar *data = this->AddCellDataOffset(&residual, i_cell);
149  const auto &integrator = cell.integrator();
150  for (int q = 0, n = integrator.CountPoints(); q < n; ++q) {
151  auto scale = 1.0 / GetWeight(integrator, q);
152  data = cell.polynomial().ScaleValueAt(scale, data);
153  }
154  assert(data == residual.data() + residual.size()
155  || data == this->AddCellDataOffset(&residual, i_cell + 1));
156  }
157  return residual;
158  }

161  void AddFluxDivergence(Cell const &cell, Scalar *residual) const override {
162  assert(residual);
163  const auto &integrator = cell.integrator();
164  for (int q = 0, n = integrator.CountPoints(); q < n; ++q) {
165  auto flux = GetWeightedFluxMatrix(cell, q);
166  auto const &grad = GetBasisGradients(cell.polynomial(), q);
167  Coeff prod = flux * grad;
168  Polynomial::AddToResidual(prod, residual);
169  }

171  void AddFluxToHolderAndSharer(Face const &face,
172  Scalar *holder_data, Scalar *sharer_data) const override {
173  const auto &riemanns = this->GetRiemannSolvers(face);
174  const auto &integrator = face.integrator();
175  const auto &holder = face.holder();
176  const auto &sharer = face.sharer();
177  auto &i_node_on_holder = i_node_on_holder_[face.id()];
178  auto &i_node_on_sharer = i_node_on_sharer_[face.id()];
179  for (int f = 0, n = integrator.CountPoints(); f < n; ++f) {
180  auto c_holder = i_node_on_holder[f];
181  auto c_sharer = i_node_on_sharer[f];
182  Value u_holder = holder.polynomial().GetValue(c_holder);
183  Value u_sharer = sharer.polynomial().GetValue(c_sharer);
184  Value flux = riemanns[f].GetFluxUpwind(u_holder, u_sharer);
185  flux *= integrator.GetGlobalWeight(f);
186  assert(holder_data);
187  holder.polynomial().MinusValue(flux, holder_data, c_holder);
188  if (nullptr == sharer_data) { continue; }
189  sharer.polynomial().AddValueTo(flux, sharer_data, c_sharer);
190  }
